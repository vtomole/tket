41 Program::BlockIterator &Program::BlockIterator::operator++() {
42  while (!stack_.empty()) {
43  FGVert last_branch = stack_.back();
44  FGVertVec succs = prog_->get_successors(last_branch);
45  for (const FGVert &s : succs) {
46  if (visited_.find(s) == visited_.end() && s != prog_->exit_) {
47  current_vert_ = s;

53  stack_.pop_back();
54  }

79 Program::CommandIterator &Program::CommandIterator::operator++() {
93  if (stage_ == ComItStage::Final) {
94  // Just output STOP; move to end()
95  *this = CommandIterator();
96  return *this;
97  }
98  while (current_block_ != prog_->block_end()) {
99  switch (stage_) {
100  case ComItStage::Label: {
101  // Just reached a new block; possibly add a label
102  stage_ = ComItStage::FirstCommand;
103  FGVert block = *current_block_;
104  FGEdgeVec ins = prog_->get_in_edges(block);
105  if (ins.size() != 1 || prog_->get_source(ins.front()) != prev_block_ ||
106  prog_->get_branch(ins.front())) {
107  // Add LABEL
108  std::string label = get_label(block);
109  Op_ptr op = std::make_shared<FlowOp>(OpType::Label, label);
110  current_command_ = Command(op, {});
111  return *this;
112  } else
113  continue;
114  }
115  case ComItStage::FirstCommand: {
116  // Just output a label; start iterating through commands of block
117  current_com_iterator_ = current_block_.get_circuit_ref().begin();
118  if (current_com_iterator_ == current_block_.get_circuit_ref().end()) {
119  stage_ = ComItStage::Branch;
120  continue;
121  } else {
122  stage_ = ComItStage::Command;
123  current_command_ = *current_com_iterator_;
124  return *this;
125  }
126  }
127  case ComItStage::Command: {
128  // Just output a command; continue
129  ++current_com_iterator_;
130  if (current_com_iterator_ == current_block_.get_circuit_ref().end()) {
131  stage_ = ComItStage::Branch;
132  continue;
133  } else {
134  current_command_ = *current_com_iterator_;
135  return *this;
136  }
137  }
138  case ComItStage::Branch: {
139  // Finished commands for block; possibly add a branch
140  FGVert block = *current_block_;
141  std::optional<Bit> condition = prog_->get_condition(block);
142  stage_ = ComItStage::Goto;
143  if (condition) {
144  // Add BRANCH
145  FGVert target = prog_->get_branch_successor(block, true);
146  std::string label = get_label(target);
147  Op_ptr op = std::make_shared<FlowOp>(OpType::Branch, label);
148  current_command_ = Command(op, {*condition});
149  return *this;
150  } else
151  continue;
152  }
153  case ComItStage::Goto: {
154  // Reached end of block; possibly add a Goto before moving to next block
155  prev_block_ = *current_block_;
156  ++current_block_;
157  stage_ = ComItStage::Label;
158  FGVert target = prog_->get_branch_successor(prev_block_, false);
159  if (current_block_ == prog_->block_end()) {
160  if (target == prog_->exit_) continue;
161  } else if (target == *current_block_)
162  continue;
163  // Add Goto
164  std::string label = get_label(target);
165  Op_ptr op = std::make_shared<FlowOp>(OpType::Goto, label);
166  current_command_ = Command(op, {});
167  return *this;
168  }
169  default: {
170  // None of Stop, Final, or End should be hit before
171  // the block iterator reaches end
172  throw ProgramError(
173  "Error in command iteration: hit final stages before "
174  "reaching exit block");
175  }
176  }
177  }
178  // Label to STOP
179  if (stage_ == ComItStage::Label) {
180  std::map<FGVert, std::string>::iterator found = labels_.find(prog_->exit_);
181  if (found != labels_.end()) {
182  Op_ptr op = std::make_shared<FlowOp>(OpType::Label, found->second);
183  current_command_ = Command(op, {});
184  stage_ = ComItStage::Stop;
185  return *this;
186  }
187  }
188  // STOP command
189  Op_ptr op = std::make_shared<FlowOp>(OpType::Stop);
190  current_command_ = Command(op, {});
191  stage_ = ComItStage::Final;
192  return *this;
193 }
