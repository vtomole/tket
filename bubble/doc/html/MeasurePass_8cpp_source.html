19 Transform Transform::delay_measures() {
20  return Transform([](Circuit &circ) {
21  bool success = false;
22  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
23  if (circ.get_OpType_from_Vertex(v) == OpType::Measure) {
24  Edge c_out_edge = circ.get_nth_out_edge(v, 1);
25  if (!circ.detect_final_Op(circ.target(c_out_edge)) ||
26  circ.n_out_edges_of_type(v, EdgeType::Boolean) != 0) {
27  throw CircuitInvalidity(
28  "Cannot commute Measure through classical operations to the end "
29  "of the circuit");
30  }
31  Edge out_edge = circ.get_nth_out_edge(v, 0);
32  Edge current_edge = out_edge;
33  Vertex current_vertex = circ.target(current_edge);
34  port_t current_port = circ.get_target_port(current_edge);
35  Op_ptr current_op = circ.get_Op_ptr_from_Vertex(current_vertex);
36  OpType current_optype = current_op->get_type();
37  while (!is_final_q_type(current_optype) &&
38  (current_optype == OpType::SWAP ||
39  current_op->commutes_with_basis(Pauli::Z, current_port))) {
40  if (current_optype == OpType::SWAP) {
41  // Update from SWAP
42  current_edge =
43  circ.get_nth_out_edge(current_vertex, 1 - current_port);
44  } else {
45  // Update to successor
46  current_edge = circ.get_nth_out_edge(current_vertex, current_port);
47  }
48  current_vertex = circ.target(current_edge);
49  current_port = circ.get_target_port(current_edge);
50  current_op = circ.get_Op_ptr_from_Vertex(current_vertex);
51  current_optype = current_op->get_type();
52  }
53  // If we haven't moved it to an output, we can't continue
54  if (!is_final_q_type(current_optype)) {
55  throw CircuitInvalidity(
56  "Cannot commute Measure through quantum gates to the end of the "
57  "circuit");
58  }
59  // If the measure was already at an output, nothing to do
60  if (current_edge == out_edge) continue;
61  Edge in_edge = circ.get_nth_in_edge(v, 0);
62  // Rewire measure
63  circ.add_edge(
64  {circ.source(in_edge), circ.get_source_port(in_edge)},
65  {circ.target(out_edge), circ.get_target_port(out_edge)},
66  EdgeType::Quantum);
67  circ.remove_edge(in_edge);
68  circ.remove_edge(out_edge);
69  circ.add_edge(
70  {circ.source(current_edge), circ.get_source_port(current_edge)},
71  {v, 0}, EdgeType::Quantum);
72  circ.add_edge({v, 0}, {current_vertex, 0}, EdgeType::Quantum);
73  circ.remove_edge(current_edge);
74  success = true;
75  }
76  }
77  return success;
78  });
79 }
