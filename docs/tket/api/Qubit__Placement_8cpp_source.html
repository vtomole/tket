28 std::set<Qubit> interacting_qbs(const Circuit& circ) {
29  std::set<Qubit> qbs;
30  for (const Qubit& qb : circ.all_qubits()) {
31  Edge e = circ.get_nth_out_edge(circ.get_in(qb), 0);
32  Vertex terminal = circ.target(circ.skip_irrelevant_edges(e));
33  if (!circ.detect_final_Op(terminal)) {
34  qbs.insert(qb);
35  }
36  }
37 
38  return qbs;
39 }

41 QubitGraph monomorph_interaction_graph(
42  const Circuit& circ, const unsigned max_edges, unsigned depth_limit) {
43  std::set<Qubit> qubits_considered = interacting_qbs(circ);
44 
45  QubitGraph q_graph(circ.all_qubits());
46 
47  RoutingFrontier current_sf(circ);
48  unsigned count_edges = 0;
49  for (unsigned slice = 0;
50  slice < depth_limit && count_edges < max_edges &&
51  !current_sf.slice->empty() && qubits_considered.size() > 1;
52  slice++) {
53  for (const Vertex& vert : *current_sf.slice) {
54  EdgeVec q_out_edges = circ.get_out_edges_of_type(vert, EdgeType::Quantum);
55  Qubit qb1;
56  Qubit qb2;
57  for (const std::pair<UnitID, Edge>& pair :
58  current_sf.quantum_out_edges->get<TagKey>()) {
59  if (pair.second == q_out_edges[0])
60  qb1 = Qubit(pair.first);
61  else if (pair.second == q_out_edges[1])
62  qb2 = Qubit(pair.first);
63  }
64  if (!q_graph.connection_exists(qb1, qb2) &&
65  !q_graph.connection_exists(qb2, qb1)) {
66  q_graph.add_connection(qb1, qb2, slice + 1);
67  count_edges++;
68  }
69  }
70 
71  current_sf.next_slicefrontier();
72  }
73  q_graph.remove_stray_uids();
74  return q_graph;
75 }

77 QubitGraph generate_interaction_graph(
78  const Circuit& circ, unsigned depth_limit) {
79  std::set<Qubit> qubits_considered = interacting_qbs(circ);
80  QubitGraph q_graph(circ.all_qubits());
81  RoutingFrontier current_sf(circ);
82 
83  for (unsigned slice = 0; slice < depth_limit && !current_sf.slice->empty() &&
84  qubits_considered.size() > 1;
85  slice++) {
86  for (const Vertex& vert : *current_sf.slice) {
87  EdgeVec q_out_edges = circ.get_out_edges_of_type(vert, EdgeType::Quantum);
88  Qubit qb1;
89  Qubit qb2;
90  for (const std::pair<UnitID, Edge>& pair :
91  current_sf.quantum_out_edges->get<TagKey>()) {
92  if (pair.second == q_out_edges[0])
93  qb1 = Qubit(pair.first);
94  else if (pair.second == q_out_edges[1])
95  qb2 = Qubit(pair.first);
96  }
97  const bool qb1_considered =
98  (qubits_considered.find(qb1) != qubits_considered.end());
99  const bool qb2_considered =
100  (qubits_considered.find(qb2) != qubits_considered.end());
101  if ((qb2 != qb1) && (qb1_considered || qb2_considered)) {
102  if (!qb1_considered) {
103  qubits_considered.erase(qb2);
104  } else if (!qb2_considered) {
105  qubits_considered.erase(qb1);
106  } else if (!q_graph.connection_exists(qb1, qb2)) {
107  const unsigned out1 = q_graph.get_degree(qb1);
108  const unsigned out2 = q_graph.get_degree(qb2);
109  q_graph.add_connection(qb1, qb2, slice + 1);
110  if (out1 == 1) {
111  qubits_considered.erase(qb1);
112  }
113  if (out2 == 1) {
114  qubits_considered.erase(qb2);
115  }
116  } else {
117  qubits_considered.erase(qb1);
118  qubits_considered.erase(qb2);
119  }
120  }
121  }
122  current_sf.next_slicefrontier();
123  }
124  q_graph.remove_stray_uids();
125 
126  return q_graph;
127 }

129 QubitLineList qubit_lines(const Circuit& circ) {
130  const QubitGraph q_graph = generate_interaction_graph(circ);
131  std::set<Qubit> all_qb;
132  for (const Qubit& qb : circ.all_qubits()) {
133  all_qb.insert(qb);
134  }
135  QubitLineList found_lines;
136  unsigned found_line_size = 0;
137  QubitGraph::UndirectedConnGraph graph = q_graph.get_undirected_connectivity();
138  do {
139  auto u_line = graphs::longest_simple_path(graph);
140  QubitLine found;
141  std::transform(
142  u_line.begin(), u_line.end(), std::back_inserter(found),
143  [&graph](auto v) -> Qubit { return graph[v].uid; });
144  found_line_size = found.size();
145  if (found_line_size > 1) {
146  found_lines.push_back(found);
147  for (const auto& vertex : u_line) {
148  boost::clear_vertex(vertex, graph);
149  }
150  for (Qubit q : found) {
151  all_qb.erase(q);
152  }
153  }
154  } while (found_line_size > 1);
155 
156  for (const Qubit& qb : circ.all_qubits()) {
157  if (all_qb.find(qb) != all_qb.end()) found_lines.push_back({qb});
158  }
159  // print_qubitlines(found_lines);
160  return found_lines;
161 }

163 // remove a given number of nodes from the architecture, return a set of usable
164 // nodes (the remainder)
165 node_set_t best_nodes(Architecture& arc, unsigned n_remove) {
166  node_set_t all_nodes = arc.get_all_uids_set();
167  node_set_t bad_nodes;
168  // if there are nodes 'removed' already count them as bad nodes
169  for (Node n : all_nodes) {
170  if (arc.get_degree(n) == 0) {
171  bad_nodes.insert(n);
172  n_remove--;
173  }
174  }
175  node_set_t removed_nodes = arc.remove_worst_nodes(n_remove);
176  bad_nodes.insert(removed_nodes.begin(), removed_nodes.end());
177  node_set_t good_nodes; // keep track of nodes of architecture actually used
178  std::set_difference(
179  all_nodes.begin(), all_nodes.end(), bad_nodes.begin(), bad_nodes.end(),
180  std::inserter(good_nodes, good_nodes.begin()));
181  return good_nodes;
182 }

184 // map qubit lines to node lines, erasing qubit lines as it goes
185 qubit_mapping_t map_lines(
186  QubitLineList& qb_lines, const std::vector<node_vector_t>& node_lines) {
187  qubit_mapping_t outmap;
188 
189  // go through all node lines
190  for (unsigned i = 0; i < node_lines.size(); i++) {
191  node_vector_t node_lst = node_lines[i];
192  for (const Node& node : node_lst) {
193  outmap.insert({*qb_lines[i].begin(), node});
194  qb_lines[i].erase(qb_lines[i].begin());
195  }
196  }
197  return outmap;
198 }

200 // trivially place qubit lines on to available nodes, return map
201 qubit_mapping_t place_qubit_lines(
202  const QubitLineList& qb_lines, node_set_t available_nodes) {
203  qubit_mapping_t outmap;
204 
205  node_set_t::iterator node_it = available_nodes.begin();
206  for (const QubitLine& line : qb_lines) {
207  for (const Qubit& qb : line) {
208  if (node_it == available_nodes.end()) {
209  throw ArchitectureInvalidity("Not enough nodes to place all qubits.");
210  }
211  outmap.insert({qb, *node_it});
212  ++node_it;
213  }
214  }
215  return outmap;
216 }

218 // Determine intial qubit placement by requesting lines of architecture to place
219 // lines of qubits on
220 qubit_mapping_t lines_on_arc(
221  Architecture arc, QubitLineList qb_lines, unsigned n_qubits) {
222  unsigned difference = arc.n_uids() - n_qubits;
223  // sort from longest to shortest
224  std::sort(qb_lines.begin(), qb_lines.end(), [](QubitLine x, QubitLine y) {
225  return (x.size() > y.size());
226  });
227 
228  // get rid of one qubit lines
229  while (!qb_lines.empty() && qb_lines.back().size() < 2) {
230  difference++;
231  qb_lines.pop_back();
232  }
233 
234  // remove poorly connected nodes, up to the number not used by mapping
235  node_set_t unused_nodes = best_nodes(arc, difference);
236 
237  // find lengths required
238  std::vector<unsigned> lengths;
239  for (const QubitLine& line : qb_lines) {
240  lengths.push_back(line.size());
241  }
242 
243  // attempt to find lines of required length on architecture
244  std::vector<node_vector_t> node_lines = arc.get_lines(lengths);
245 
246  // map qubit lines to node lines to some extent
247  qubit_mapping_t outmap = map_lines(qb_lines, node_lines);
248  for (qubit_mapping_t::iterator it = outmap.begin(); it != outmap.end();
249  ++it) {
250  unused_nodes.erase(it->second);
251  }
252  // map remain remaining qubit lines trivially
253  qubit_mapping_t remainder_map = place_qubit_lines(qb_lines, unused_nodes);
254  outmap.insert(remainder_map.begin(), remainder_map.end());
255 
256  return outmap;
257 }

271 // calculate a cost value for map being considered
272 double Monomorpher::map_cost(const qubit_bimap_t& n_map) {
273  double cost = 0.0;
274  const int approx_depth = circ.n_gates() / circ.n_qubits() + 1;
275  // constants for scaling single qubit error
276  constexpr double c1 = 0.5;
277  constexpr double d1 = 1 - 1 / c1;
278  for (auto [qb, node] : n_map) {
279  // add fidelities of edges from node, weighted by if edge is used by
280  // interaction graph
281  node_set_t neighs = arc.get_neighbour_uids(node);
282  double edge_sum = 1.0;
283  for (Node nei : neighs) {
284  double fwd_edge_weighting = 1.0, bck_edge_weighting = 1.0;
285  // check if neighbour node is mapped
286  r_const_iterator_t nei_qb_it = n_map.right.find(nei);
287  if (nei_qb_it == n_map.right.end()) continue;
288  // if ( nei_qb_it != n_map.end()){
289  unsigned edge_val = q_graph.get_connection_weight(qb, nei_qb_it->second);
290  // check if either directed interaction exists
291  auto place_interactions_boost = [&](unsigned edge_v) {
292  return config.depth_limit - edge_v + 1;
293  };
294  // if edge is used by interaction in mapping, weight edge higher
295  if (edge_val) {
296  fwd_edge_weighting += place_interactions_boost(edge_val);
297  } else {
298  edge_val = q_graph.get_connection_weight(nei_qb_it->second, qb);
299  if (edge_val) {
300  bck_edge_weighting += place_interactions_boost(edge_val);
301  }
302  }
303  // }
304  gate_error_t fwd_error = characterisation.get_error({node, nei});
305  gate_error_t bck_error = characterisation.get_error({nei, node});
306  edge_sum += fwd_edge_weighting * (1.0 - fwd_error);
307  edge_sum += bck_edge_weighting * (1.0 - bck_error);
308  }
309 
310  // bigger edge sum -> smaller cost
311  cost += 1.0 / (edge_sum);
312 
313  // add error rate of node
314  gate_error_t single_error = characterisation.get_error(node);
315  cost += d1 + 1.0 / ((1.0 - single_error) + c1);
316  readout_error_t readout_error = characterisation.get_readout_error(node);
317  cost += (d1 + 1.0 / ((1.0 - readout_error) + c1)) / (approx_depth * 20);
318  // TODO add readout weighting to PlacementConfig?
319  }
320 
321  return cost;
322 }

324 std::vector<MapCost> Monomorpher::place(unsigned max_return) {
325  if (max_return < 1)
326  throw PlacementError("Max return maps for place must be at least 1.");
327  const unsigned interacting_nodes = q_graph.n_connected();
328  if ((circ.n_gates() / circ.n_qubits()) >= config.arc_contraction_ratio &&
329  circ.n_qubits() > 3) {
330  best_nodes(arc, arc.n_uids() - interacting_nodes);
331  }
332 
333  std::vector<qubit_bimap_t> potential_maps = monomorphism_edge_break(
334  arc, q_graph, config.vf2_max_matches, config.timeout);
335  std::vector<MapCost> map_costs;
336  for (unsigned i = 0; i < potential_maps.size(); i++) {
337  qubit_bimap_t& chosen = potential_maps[i];
338  double cost = map_cost(chosen);
339  qubit_mapping_t converted_map;
340  for (auto [qb, node] : chosen.left) {
341  converted_map[qb] = node;
342  }
343  map_costs.push_back({converted_map, cost});
344  }
345 
346  max_return = std::min(static_cast<unsigned>(map_costs.size()), max_return);
347  std::nth_element(
348  map_costs.begin(), map_costs.begin() + max_return - 1, map_costs.end());
349  const MapCost nth_map = *(map_costs.begin() + max_return - 1);
350  for (std::vector<MapCost>::iterator it = map_costs.begin();
351  it != map_costs.end();) {
352  if (*it > nth_map) {
353  it = map_costs.erase(it);
354  } else {
355  ++it;
356  }
357  }
358 
359  if (max_return < map_costs.size()) map_costs.resize(max_return);
360  return map_costs;
361 }
