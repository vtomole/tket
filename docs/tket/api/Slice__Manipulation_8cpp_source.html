22 RoutingFrontier::RoutingFrontier(const Circuit& _circ) : circ(_circ) { init(); }
23 void RoutingFrontier::init() {
24  VertexVec input_slice;
25  quantum_in_edges = std::make_shared<unit_frontier_t>();
26  classical_in_edges = std::make_shared<b_frontier_t>();
27 
28  for (const Qubit& qb : circ.all_qubits()) {
29  Vertex input = circ.get_in(qb);
30  input_slice.push_back(input);
31  Edge candidate = circ.get_nth_out_edge(input, 0);
32  quantum_in_edges->insert({qb, circ.skip_irrelevant_edges(candidate)});
33  }
34  for (const Bit& bit : circ.all_bits()) {
35  Vertex input = circ.get_in(bit);
36  EdgeVec candidates = circ.get_nth_b_out_bundle(input, 0);
37  classical_in_edges->insert({bit, candidates});
38  }
39 
40  CutFrontier next_cut = circ.next_cut(quantum_in_edges, classical_in_edges);
41  slice = next_cut.slice;
42  quantum_out_edges = next_cut.u_frontier;
43 }

45 void RoutingFrontier::next_slicefrontier() {
46  quantum_in_edges = std::make_shared<unit_frontier_t>();
47  classical_in_edges = std::make_shared<b_frontier_t>();
48  for (const std::pair<UnitID, Edge>& pair : quantum_out_edges->get<TagKey>()) {
49  Edge new_e = circ.skip_irrelevant_edges(pair.second);
50  quantum_in_edges->insert({pair.first, new_e});
51  Vertex targ = circ.target(new_e);
52  EdgeVec targ_classical_ins =
53  circ.get_in_edges_of_type(targ, EdgeType::Boolean);
54  classical_in_edges->insert(
55  {Bit("frontier_bit", pair.first.index()), targ_classical_ins});
56  }
57 
58  CutFrontier next_cut = circ.next_cut(quantum_in_edges, classical_in_edges);
59  slice = next_cut.slice;
60  quantum_out_edges = next_cut.u_frontier;
61 }

63 std::vector<Node> Routing::nodes_from_qubits(const qubit_vector_t& qubs) {
64  std::vector<Node> nodes;
65  unsigned start = 0;
66  if (qmap.empty()) {
67  Node node0 = *(original_arc_.max_degree_uids().begin());
68  activate_node(node0);
69  qmap.left.insert({qubs[0], node0});
70  init_map.left.insert({qubs[0], node0});
71  nodes.push_back(node0);
72  start++;
73  }
74 
75  for (unsigned i = start; i < qubs.size(); i++) {
76  l_const_iterator_t node_find = qmap.left.find(qubs[i]);
77  if (node_find == qmap.left.end()) {
78  if (i < qubs.size() - 1 &&
79  qmap.left.find(qubs[i + 1]) !=
80  qmap.left.end()) { // TODO: Could this if condition cause some
81  // nasty non determinism?
82  reactivate_qubit(qubs[i], qubs[i + 1]);
83  nodes.push_back(qmap.left.at(qubs[i]));
84  } else {
85  if (i != 0) {
86  reactivate_qubit(qubs[i], qubs[0]);
87  nodes.push_back(qmap.left.at(qubs[i]));
88  } else {
89  reactivate_qubit(qubs[i], qmap.begin()->left);
90  nodes.push_back(qmap.left.at(qubs[i]));
91  }
92  }
93  } else {
94  nodes.push_back(node_find->second);
95  }
96  }
97  return nodes;
98 }

100 /*
101 Advances slice frontier past any two_qubit operations on adjacent nodes
102 */
103 bool Routing::advance_frontier() {
104  bool found_adjacent_op = true;
105  while (found_adjacent_op && !slice_frontier_.slice->empty()) {
106  found_adjacent_op = false;
107  for (const Vertex& vert : *slice_frontier_.slice) {
108  qubit_vector_t qubs;
109  for (const Edge& q_out :
110  circ_.get_out_edges_of_type(vert, EdgeType::Quantum)) {
111  for (const std::pair<UnitID, Edge>& pair :
112  slice_frontier_.quantum_out_edges->get<TagKey>()) {
113  if (pair.second == q_out) {
114  qubs.push_back(Qubit(pair.first));
115  break;
116  }
117  }
118  }
119  // Find OpType. If OpType is a Conditional, unpack to find vertex inside.
120  // If it's nested, this will fail.
121  OpType vert_type = circ_.get_OpType_from_Vertex(vert);
122  if (vert_type == OpType::Conditional) {
123  const Conditional& b = static_cast<const Conditional&>(
124  *circ_.get_Op_ptr_from_Vertex(vert));
125  vert_type = b.get_op()->get_type();
126  }
127 
128  // the vertex must be two qubits or a bridge, which we can skip past
129 
130  if (qubs.size() != 2 && vert_type != OpType::BRIDGE &&
131  vert_type != OpType::Barrier) {
132  throw(CircuitInvalidity(
133  "Vertex has " + std::to_string(qubs.size()) +
134  " qubits, expected 2."));
135  }
136  // BRIDGE gates are guaranteed to be across 3 adjacent nodes,
137  // already mapped so can just be read directly from the qmap
138  // otherwise, qubits may need to be activated first
139  std::vector<Node> nods = nodes_from_qubits(qubs);
140 
141  bool all_qbs_adjacent = true;
142  for (unsigned i = 0; i < nods.size() - 1; i++) {
143  all_qbs_adjacent &=
144  (current_arc_.get_distance(nods[i], nods[i + 1]) == 1);
145  }
146  if (all_qbs_adjacent ||
147  vert_type == OpType::Barrier) { // if by eachother
148  found_adjacent_op = true; // i.e. at least one 2qb gate has
149  // been able to run
150  // for all qubits skip subsequent single qubit vertices to move
151  // in edges to be prior to next multiqubit vertex
152  for (const Qubit& qub : qubs) {
153  Edge new_e = circ_.skip_irrelevant_edges(
154  slice_frontier_.quantum_out_edges->find(qub)->second);
155  slice_frontier_.quantum_in_edges->replace(
156  slice_frontier_.quantum_in_edges->find(qub), {qub, new_e});
157  Vertex targ = circ_.target(new_e);
158  EdgeVec targ_classical_ins =
159  circ_.get_in_edges_of_type(targ, EdgeType::Boolean);
160  Bit b("frontier_bit", qub.index());
161  if (slice_frontier_.classical_in_edges->find(b) ==
162  slice_frontier_.classical_in_edges->end()) {
163  slice_frontier_.classical_in_edges->insert({b, targ_classical_ins});
164  } else {
165  slice_frontier_.classical_in_edges->replace(
166  slice_frontier_.classical_in_edges->find(b),
167  {b, targ_classical_ins});
168  }
169  }
170  }
171  }
172  if (found_adjacent_op) {
173  CutFrontier next_cut = circ_.next_cut(
174  slice_frontier_.quantum_in_edges, slice_frontier_.classical_in_edges);
175  slice_frontier_.slice = next_cut.slice;
176  slice_frontier_.quantum_out_edges = next_cut.u_frontier;
177  slice_frontier_.classical_in_edges = std::make_shared<b_frontier_t>();
178  for (const std::pair<UnitID, Edge>& pair :
179  slice_frontier_.quantum_in_edges->get<TagKey>()) {
180  Vertex targ = circ_.target(pair.second);
181  EdgeVec targ_classical_ins =
182  circ_.get_in_edges_of_type(targ, EdgeType::Boolean);
183  Bit b("frontier_bit", pair.first.index());
184  slice_frontier_.classical_in_edges->insert({b, targ_classical_ins});
185  }
186  }
187  }
188 
189  interaction = generate_interaction_frontier(slice_frontier_); // reset
190  dist_vector = generate_distance_vector(interaction);
191  return found_adjacent_op;
192 }

194 Interactions Routing::generate_interaction_frontier(
195  const RoutingFrontier& slice_front) {
196  Interactions inter;
197  for (const UnitID& uid : current_arc_.get_all_uids()) {
198  Node n(uid);
199  inter.insert({n, n});
200  }
201  for (const Vertex& vert : *slice_front.slice) {
202  qubit_vector_t qubs;
203  for (const Edge& q_out :
204  circ_.get_out_edges_of_type(vert, EdgeType::Quantum)) {
205  for (const std::pair<UnitID, Edge>& pair :
206  slice_front.quantum_out_edges->get<TagKey>()) {
207  if (pair.second == q_out) {
208  qubs.push_back(Qubit(pair.first));
209  break;
210  }
211  }
212  }
213  // if generate_interaction_frontier called with slice_frontier_ no ops with
214  // more than two qubits will be present if generate_interaction_frontier
215  // called with frontier made in try_all_swaps or check_distributed_cx,
216  // Barrier Op possible. If barrier op in slice, don't add qubits in barrier
217  // interaction to Interactions
218  if (qubs.size() != 2) {
219  if (circ_.get_OpType_from_Vertex(vert) == OpType::Barrier) continue;
220  throw CircuitInvalidity(
221  "Vertex has " + std::to_string(qubs.size()) + " qubits, expected 2.");
222  }
223 
224  l_const_iterator_t node0_find = qmap.left.find(qubs[0]);
225  l_const_iterator_t node1_find = qmap.left.find(qubs[1]);
226  if (node0_find != qmap.left.end() && node1_find != qmap.left.end()) {
227  Node one = node0_find->second;
228  Node two = node1_find->second;
229  inter[one] = two;
230  inter[two] = one;
231  }
232  }
233  return inter;
234 }
