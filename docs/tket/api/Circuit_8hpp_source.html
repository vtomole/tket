1536 bool Circuit::rename_units(const std::map<UnitA, UnitB> &qm) {
1537  // Can only work for Unit classes
1538  static_assert(std::is_base_of<UnitID, UnitA>::value);
1539  static_assert(std::is_base_of<UnitID, UnitB>::value);
1540  // Unit types must be related, so cannot rename e.g. Bits to Qubits
1541  static_assert(
1542  std::is_base_of<UnitA, UnitB>::value ||
1543  std::is_base_of<UnitB, UnitA>::value);
1544  std::map<UnitID, BoundaryElement> new_elems;
1545  bool modified = false;
1546  for (const std::pair<const UnitA, UnitB> &pair : qm) {
1547  boundary_t::iterator found = boundary.get<TagID>().find(pair.first);
1548  if (found == boundary.get<TagID>().end()) {
1549  tket_log()->warn("unit {} not found in circuit", pair.first.repr());
1550  continue;
1551  }
1552 
1553  opt_reg_info_t target_reg_info = get_reg_info(pair.second.reg_name());
1554  if (target_reg_info) {
1555  if (target_reg_info.value().first != found->type())
1556  throw CircuitInvalidity(
1557  "Incompatible registers: " + pair.first.reg_name() + " and " +
1558  pair.second.reg_name());
1559  if (target_reg_info.value().second != pair.second.reg_dim())
1560  throw CircuitInvalidity(
1561  "Existing register " + pair.second.reg_name() +
1562  " cannot support id: " + pair.second.repr());
1563  }
1564  std::pair<std::map<UnitID, BoundaryElement>::iterator, bool> inserted =
1565  new_elems.insert({pair.second, {pair.second, found->in_, found->out_}});
1566  if (!inserted.second)
1567  throw CircuitInvalidity(
1568  "Mapping two units to the same id: " + pair.second.repr());
1569  modified = true;
1570  boundary.erase(found);
1571  }
1572  for (const std::pair<const UnitID, BoundaryElement> &pair : new_elems) {
1573  std::pair<boundary_t::iterator, bool> added = boundary.insert(pair.second);
1574  if (!added.second)
1575  throw CircuitInvalidity(
1576  "Unit already exists in circuit: " + pair.first.repr());
1577  TKET_ASSERT(modified);
1578  }
1579  update_initial_map(qm);
1580  update_final_map(qm);
1581  return modified;
1582 }
