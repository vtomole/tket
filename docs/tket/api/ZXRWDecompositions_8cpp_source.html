12  }

14  while (!to_decompose.empty()) {

18  const ZXDiagram& inner = *zxb.get_diagram();
19  // Substitute diagram in place of box
20  auto iso = diag.copy_graph(inner, false);
21  std::map<Wire, Wire> incident_map;
22  for (const Wire& w : diag.adj_wires(box)) {
23  WireProperties wp = diag.get_wire_info(w);
24  ZXVert s = diag.source(w);
25  ZXVert t = diag.target(w);
26  if (s == box) {

28  Wire bw = diag.adj_wires(b).at(0);
29  if (diag.end_of(bw, b) == WireEnd::Source) {
30  s = diag.target(bw);
31  wp.source_port = wp.target_port;
32  } else {
33  s = diag.source(bw);
34  wp.source_port = wp.source_port;
35  }
36  }
37  if (t == box) {
38  ZXVert b = iso.first.at(inner.boundary.at(*wp.target_port));
39  Wire bw = diag.adj_wires(b).at(0);
40  if (diag.end_of(bw, b) == WireEnd::Source) {

43  } else {
44  t = diag.source(bw);
45  wp.target_port = wp.source_port;
46  }
47  }

49  incident_map.emplace(w, new_w);
50  }

52  diag.remove_vertex(box);
53  for (const std::pair<const ZXVert, ZXVert>& pair : iso.first) {

55  if (is_boundary_type(type))

59  }
60  }
