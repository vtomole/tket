108 void fill_partial_mapping(
109  const qubit_vector_t &current_qubits, qubit_mapping_t &partial_mapping) {
110  unsigned up_nu = 0;
111  for (Qubit q : current_qubits) {
112  if (partial_mapping.find(q) == partial_mapping.end()) {
113  partial_mapping.insert({q, Node(Placement::unplaced_reg(), up_nu)});
114  up_nu++;
115  }
116  }
117 
118  if (std::any_of(
119  partial_mapping.begin(), partial_mapping.end(),
120  [&current_qubits](auto x) {
121  return std::find(
122  current_qubits.begin(), current_qubits.end(), x.first) ==
123  current_qubits.end();
124  })) {
125  tket_log()->warn("mapping contains qubits not present in circuit");
126  }
127 }

177 std::vector<qubit_mapping_t> GraphPlacement::get_all_placement_maps(
178  const Circuit &circ_) const {
179  QubitGraph q_graph = monomorph_interaction_graph(
180  circ_, arc_.n_connections(), config_.depth_limit);
181  std::vector<qubit_bimap_t> all_bimaps = monomorphism_edge_break(
182  arc_, q_graph, config_.vf2_max_matches, config_.timeout);
183  std::vector<qubit_mapping_t> all_qmaps;
184  qubit_vector_t all_qbs = circ_.all_qubits();
185  for (qubit_bimap_t bm : all_bimaps) {
186  qubit_mapping_t qm = bimap_to_map(bm.left);
187  fill_partial_mapping(all_qbs, qm);
188  all_qmaps.push_back(qm);
189  }
190  return all_qmaps;
191 }

198 std::vector<qubit_mapping_t> NoiseAwarePlacement::get_all_placement_maps(
199  const Circuit &circ_) const {
200  // Set up default placements configuation
201  // Place according to configuration
202  Monomorpher placer(circ_, arc_, characterisation_, config_);
203  std::vector<MapCost> results = placer.place(config_.depth_limit * 2);
204  std::sort(results.begin(), results.end());
205  std::vector<qubit_mapping_t> output;
206  qubit_vector_t all_qbs = circ_.all_qubits();
207  for (const MapCost &map_c : results) {
208  qubit_mapping_t map_ = map_c.map;
209  fill_partial_mapping(all_qbs, map_);
210  output.push_back(map_c.map);
211  }
212  return output;
213 }
