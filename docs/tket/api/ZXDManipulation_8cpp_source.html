130 void ZXDiagram::check_validity() const {
131  std::set<ZXVert> boundary_lookup;
132  for (const ZXVert& b : boundary) {
133  if (!is_boundary_type(get_zxtype(b)))
134  throw ZXError("Non-boundary vertex type in boundary");
135  if (!boundary_lookup.insert(b).second)
136  throw ZXError("Vertex appears in boundary multiple times");
137  }
138  BGL_FORALL_VERTICES(v, *graph, ZXGraph) {
139  ZXGen_ptr op = get_vertex_ZXGen_ptr(v);
140  ZXType type = op->get_type();
141  if (is_boundary_type(type)) {
142  if (degree(v) != 1)
143  throw ZXError("Boundary vertex does not have degree 1");
144  if (boundary_lookup.find(v) == boundary_lookup.end())
145  throw ZXError("Vertex of boundary type is not in the boundary");
146  }
147  std::optional<unsigned> n_ports = std::nullopt;
148  if (is_directed_type(type)) {
149  const ZXDirected& dir = static_cast<const ZXDirected&>(*op);
150  n_ports = dir.n_ports();
151  }
152  std::vector<bool> ports_found(n_ports ? *n_ports : 0, false);
153  BGL_FORALL_OUTEDGES(v, w, *graph, ZXGraph) {
154  check_valid_wire(source_port(w), get_qtype(w), n_ports, ports_found, op);
155  }
156  BGL_FORALL_INEDGES(v, w, *graph, ZXGraph) {
157  check_valid_wire(target_port(w), get_qtype(w), n_ports, ports_found, op);
158  }
159  if (n_ports && !std::all_of(
160  ports_found.begin(), ports_found.end(),
161  [](const bool& b) { return b; }))
162  throw ZXError("Not all ports of a directed vertex have wires connected");
163  }
164 }
