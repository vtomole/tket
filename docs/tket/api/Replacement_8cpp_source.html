23 Circuit CX_circ_from_multiq(const Op_ptr op) {
24  OpDesc desc = op->get_desc();
25  if (!desc.is_gate())
26  throw NotImplemented(
27  "Can only build replacement circuits for basic gates; given " +
28  desc.name());
29  unsigned n_qubits = op->n_qubits();
30  switch (desc.type()) {
31  case OpType::CZ: {
32  return CircPool::CZ_using_CX();
33  }
34  case OpType::CY: {
35  return CircPool::CY_using_CX();
36  }
37  case OpType::CH: {
38  return CircPool::CH_using_CX();
39  }
40  case OpType::CV: {
41  return CircPool::CV_using_CX();
42  }
43  case OpType::CVdg: {
44  return CircPool::CVdg_using_CX();
45  }
46  case OpType::CSX: {
47  return CircPool::CSX_using_CX();
48  }
49  case OpType::CSXdg: {
50  return CircPool::CSXdg_using_CX();
51  }
52  case OpType::CRz: {
53  return CircPool::CRz_using_CX(op->get_params()[0]);
54  }
55  case OpType::CRx: {
56  return CircPool::CRx_using_CX(op->get_params()[0]);
57  }
58  case OpType::CRy: {
59  return CircPool::CRy_using_CX(op->get_params()[0]);
60  }
61  case OpType::CU1: {
62  return CircPool::CU1_using_CX(op->get_params()[0]);
63  }
64  case OpType::CU3: {
65  std::vector<Expr> params = op->get_params();
66  return CircPool::CU3_using_CX(params[0], params[1], params[2]);
67  }
68  case OpType::SWAP: {
69  return CircPool::SWAP_using_CX_0();
70  }
71  case OpType::CSWAP: {
72  return CircPool::CSWAP_using_CX();
73  }
74  case OpType::PhaseGadget: {
75  return phase_gadget(n_qubits, op->get_params()[0], CXConfigType::Snake);
76  }
77  case OpType::ISWAP: {
78  return CircPool::ISWAP_using_CX(op->get_params()[0]);
79  }
80  case OpType::XXPhase: {
81  return CircPool::XXPhase_using_CX(op->get_params()[0]);
82  }
83  case OpType::ECR: {
84  return CircPool::ECR_using_CX();
85  }
86  case OpType::ZZMax: {
87  return CircPool::ZZMax_using_CX();
88  }
89  case OpType::ZZPhase: {
90  return CircPool::ZZPhase_using_CX(op->get_params()[0]);
91  }
92  case OpType::YYPhase: {
93  return CircPool::YYPhase_using_CX(op->get_params()[0]);
94  }
95  case OpType::XXPhase3: {
96  return CircPool::XXPhase3_using_CX(op->get_params()[0]);
97  }
98  case OpType::BRIDGE: {
99  return CircPool::BRIDGE_using_CX_0();
100  }
101  case OpType::CnRy: {
102  return Transform::decomposed_CnRy(op, n_qubits);
103  }
104  case OpType::CCX: {
105  return CircPool::CCX_normal_decomp();
106  }
107  case OpType::CnX: {
108  return Transform::cnx_normal_decomp(n_qubits - 1);
109  }
110  case OpType::ESWAP: {
111  return CircPool::ESWAP_using_CX(op->get_params()[0]);
112  }
113  case OpType::FSim: {
114  return CircPool::FSim_using_CX(op->get_params()[0], op->get_params()[1]);
115  }
116  case OpType::Sycamore: {
117  return CircPool::FSim_using_CX(0.5, 1. / 6.);
118  }
119  case OpType::ISWAPMax: {
120  return CircPool::ISWAP_using_CX(1.);
121  }
122  case OpType::PhasedISWAP: {
123  return CircPool::PhasedISWAP_using_CX(
124  op->get_params()[0], op->get_params()[1]);
125  }
126  case OpType::NPhasedX: {
127  std::vector<Expr> params = op->get_params();
128  return CircPool::NPhasedX_using_CX(n_qubits, params[0], params[1]);
129  }
130  default: {
131  throw NotImplemented(
132  "Cannot find replacement circuit for OpType::" + desc.name());
133  }
134  }
135 }

137 Circuit CX_ZX_circ_from_op(const Op_ptr op) {
138  OpDesc desc = op->get_desc();
139  if (!desc.is_gate())
140  throw NotImplemented(
141  "Can only build replacement circuits for basic gates; given " +
142  desc.name());
143  switch (desc.type()) {
144  case OpType::Z: {
145  Circuit replacement(1);
146  replacement.add_op<unsigned>(OpType::Rz, 1., {0});
147  replacement.add_phase(0.5);
148  return replacement;
149  }
150  case OpType::X: {
151  Circuit replacement(1);
152  replacement.add_op<unsigned>(OpType::Rx, 1., {0});
153  replacement.add_phase(0.5);
154  return replacement;
155  }
156  case OpType::Y: {
157  Circuit replacement(1);
158  replacement.add_op<unsigned>(OpType::Rz, 1., {0});
159  replacement.add_op<unsigned>(OpType::Rx, 1., {0});
160  replacement.add_phase(-0.5);
161  return replacement;
162  }
163  case OpType::S: {
164  Circuit replacement(1);
165  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
166  replacement.add_phase(0.25);
167  return replacement;
168  }
169  case OpType::Sdg: {
170  Circuit replacement(1);
171  replacement.add_op<unsigned>(OpType::Rz, -0.5, {0});
172  replacement.add_phase(-0.25);
173  return replacement;
174  }
175  case OpType::T: {
176  Circuit replacement(1);
177  replacement.add_op<unsigned>(OpType::Rz, 0.25, {0});
178  replacement.add_phase(0.125);
179  return replacement;
180  }
181  case OpType::Tdg: {
182  Circuit replacement(1);
183  replacement.add_op<unsigned>(OpType::Rz, -0.25, {0});
184  replacement.add_phase(-0.125);
185  return replacement;
186  }
187  case OpType::V: {
188  Circuit replacement(1);
189  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
190  return replacement;
191  }
192  case OpType::Vdg: {
193  Circuit replacement(1);
194  replacement.add_op<unsigned>(OpType::Rx, -0.5, {0});
195  return replacement;
196  }
197  case OpType::SX: {
198  Circuit replacement(1);
199  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
200  replacement.add_phase(0.25);
201  return replacement;
202  }
203  case OpType::SXdg: {
204  Circuit replacement(1);
205  replacement.add_op<unsigned>(OpType::Rx, -0.5, {0});
206  replacement.add_phase(-0.25);
207  return replacement;
208  }
209  case OpType::H: {
210  Circuit replacement(1);
211  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
212  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
213  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
214  replacement.add_phase(0.5);
215  return replacement;
216  }
217  case OpType::Ry: {
218  Expr angle = op->get_params()[0];
219  Circuit replacement(1);
220  replacement.add_op<unsigned>(OpType::Rz, -0.5, {0});
221  replacement.add_op<unsigned>(OpType::Rx, angle, {0});
222  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
223  return replacement;
224  }
225  case OpType::Rx:
226  case OpType::Rz:
227  case OpType::Measure:
228  case OpType::Collapse: {
229  Circuit replacement(1);
230  replacement.add_op<unsigned>(op, {0});
231  return replacement;
232  }
233  case OpType::U3: {
234  Circuit replacement(1);
235  Expr angle_z1 = op->get_params()[2];
236  Expr angle_y = op->get_params()[0];
237  Expr angle_z2 = op->get_params()[1];
238  replacement.add_op<unsigned>(OpType::Rz, angle_z1 - 0.5, {0});
239  replacement.add_op<unsigned>(OpType::Rx, angle_y, {0});
240  replacement.add_op<unsigned>(OpType::Rz, angle_z2 + 0.5, {0});
241  replacement.add_phase((angle_z1 + angle_z2) / 2);
242  return replacement;
243  }
244  case OpType::U2: {
245  Circuit replacement(1);
246  Expr angle_z1 = op->get_params()[1];
247  Expr angle_z2 = op->get_params()[0];
248  replacement.add_op<unsigned>(OpType::Rz, angle_z1 - 0.5, {0});
249  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
250  replacement.add_op<unsigned>(OpType::Rz, angle_z2 + 0.5, {0});
251  replacement.add_phase((angle_z1 + angle_z2) / 2);
252  return replacement;
253  }
254  case OpType::U1: {
255  Circuit replacement(1);
256  Expr angle = op->get_params()[0];
257  replacement.add_op<unsigned>(OpType::Rz, angle, {0});
258  replacement.add_phase(angle / 2);
259  return replacement;
260  }
261  case OpType::PhasedX: {
262  Circuit replacement(1);
263  Expr theta = op->get_params()[0];
264  Expr phi = op->get_params()[1];
265  replacement.add_op<unsigned>(OpType::Rz, -phi, {0});
266  replacement.add_op<unsigned>(OpType::Rx, theta, {0});
267  replacement.add_op<unsigned>(OpType::Rz, phi, {0});
268  return replacement;
269  }
270  case OpType::CX: {
271  Circuit replacement(2);
272  replacement.add_op<unsigned>(op, {0, 1});
273  return replacement;
274  }
275  case OpType::CY:
276  case OpType::CZ:
277  case OpType::CH:
278  case OpType::CV:
279  case OpType::CVdg:
280  case OpType::CSX:
281  case OpType::CSXdg:
282  case OpType::CRz:
283  case OpType::CRx:
284  case OpType::CRy:
285  case OpType::CU1:
286  case OpType::CU3:
287  case OpType::PhaseGadget:
288  case OpType::CCX:
289  case OpType::SWAP:
290  case OpType::CSWAP:
291  case OpType::ECR:
292  case OpType::ISWAP:
293  case OpType::XXPhase:
294  case OpType::XXPhase3:
295  case OpType::ZZMax:
296  case OpType::ZZPhase:
297  case OpType::YYPhase:
298  case OpType::CnRy:
299  case OpType::CnX:
300  case OpType::ESWAP:
301  case OpType::FSim:
302  case OpType::Sycamore:
303  case OpType::ISWAPMax:
304  case OpType::BRIDGE: {
305  Circuit replacement = CX_circ_from_multiq(op);
306  Transform::decompose_ZX().apply(replacement);
307  return replacement;
308  }
309  case OpType::tk1: {
310  Circuit replacement(1);
311  std::vector<Expr> params = op->get_params();
312  replacement.add_op<unsigned>(OpType::Rz, params[2], {0});
313  replacement.add_op<unsigned>(OpType::Rx, params[1], {0});
314  replacement.add_op<unsigned>(OpType::Rz, params[0], {0});
315  return replacement;
316  }
317  default:
318  throw NotImplemented(
319  "Cannot find replacement circuit for OpType::" + desc.name());
320  }
321 }
