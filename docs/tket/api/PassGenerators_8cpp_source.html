30 PassPtr gen_rebase_pass(
31  const OpTypeSet& multiqs, const Circuit& cx_replacement,
32  const OpTypeSet& singleqs,
33  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
34  tk1_replacement) {
35  Transform t = Transform::rebase_factory(
36  multiqs, cx_replacement, singleqs, tk1_replacement);
37 
38  PredicatePtrMap precons;
39  OpTypeSet all_types(singleqs);
40  all_types.insert(multiqs.begin(), multiqs.end());
41  all_types.insert(OpType::Measure);
42  all_types.insert(OpType::Collapse);
43  all_types.insert(OpType::Reset);
44  PredicatePtr postcon1 = std::make_shared<GateSetPredicate>(all_types);
45  PredicatePtr postcon2 = std::make_shared<MaxTwoQubitGatesPredicate>();
46  std::pair<const std::type_index, PredicatePtr> pair2 =
47  CompilationUnit::make_type_pair(postcon1);
48  PredicatePtrMap s_postcons{pair2, CompilationUnit::make_type_pair(postcon2)};
49  PredicateClassGuarantees g_postcons{{pair2.first, Guarantee::Clear}};
50  PostConditions pc = {s_postcons, g_postcons, Guarantee::Preserve};
51  // record pass config
52  nlohmann::json j;
53  j["name"] = "RebaseCustom";
54  j["basis_multiqs"] = multiqs;
55  j["basis_cx_replacement"] = cx_replacement;
56  j["basis_singleqs"] = singleqs;
57  j["basis_tk1_replacement"] =
58  "SERIALIZATION OF FUNCTIONS IS NOT YET SUPPORTED";
59  return std::make_shared<StandardPass>(precons, t, pc, j);
60 }

185 PassPtr gen_routing_pass(const Architecture& arc, const RoutingConfig& config) {
186  Transform::Transformation trans =
187  [=](Circuit& circ) { // this doesn't work if capture by ref for some
188  // reason....
189  Routing route(circ, arc);
190  std::pair<Circuit, bool> circbool = route.solve(config);
191  circ = circbool.first;
192  return circbool.second;
193  };
194  Transform t = Transform(trans);
195 
196  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
197  PredicatePtr placedpred = std::make_shared<PlacementPredicate>(arc);
198  PredicatePtr n_qubit_pred =
199  std::make_shared<MaxNQubitsPredicate>(arc.n_uids());
200  PredicatePtrMap precons{
201  CompilationUnit::make_type_pair(placedpred),
202  CompilationUnit::make_type_pair(twoqbpred),
203  CompilationUnit::make_type_pair(n_qubit_pred)};
204 
205  PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
206  std::pair<const std::type_index, PredicatePtr> pair1 =
207  CompilationUnit::make_type_pair(postcon1);
208  PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
209  PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
210  std::type_index gateset_ti = typeid(GateSetPredicate);
211  // clears all Routing predicates apart from the ones which are specified
212  // clears all GateSet predicates (inserts SWAPs and BRIDGEs)
213  PredicateClassGuarantees g_postcons{
214  {pair1.first, Guarantee::Clear},
215  {gateset_ti, Guarantee::Clear},
216  {typeid(MaxTwoQubitGatesPredicate), Guarantee::Clear}};
217  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
218 
219  // record pass config
220  nlohmann::json j;
221  j["name"] = "RoutingPass";
222  j["routing_config"] = config;
223  j["architecture"] = arc;
224 
225  return std::make_shared<StandardPass>(precons, t, pc, j);
226 }

228 PassPtr gen_placement_pass_phase_poly(const Architecture& arc) {
229  Transform::Transformation trans = [=](Circuit& circ) {
230  if (arc.n_uids() < circ.n_qubits()) {
231  throw CircuitInvalidity(
232  "Circuit has more qubits than the architecture has nodes.");
233  }
234 
235  qubit_vector_t q_vec = circ.all_qubits();
236  std::map<Qubit, Node> qubit_to_nodes;
237  unsigned counter = 0;
238 
239  for (Node x : arc.get_all_uids_set()) {
240  if (counter < circ.n_qubits()) {
241  qubit_to_nodes.insert({q_vec[counter], x});
242  ++counter;
243  }
244  }
245 
246  circ.rename_units(qubit_to_nodes);
247 
248  return true;
249  };
250  Transform t = Transform(trans);
251 
252  PredicatePtrMap precons{};
253  PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(arc);
254  PredicatePtr n_qubit_pred =
255  std::make_shared<MaxNQubitsPredicate>(arc.n_uids());
256  PredicatePtrMap s_postcons{
257  CompilationUnit::make_type_pair(placement_pred),
258  CompilationUnit::make_type_pair(n_qubit_pred)};
259  PostConditions pc{s_postcons, {}, Guarantee::Preserve};
260  // record pass config
261  nlohmann::json j;
262  j["name"] = "PlacementPass";
263  PlacementPtr pp = std::make_shared<GraphPlacement>(arc);
264  j["params"]["placement"] = pp;
265  return std::make_shared<StandardPass>(precons, t, pc, j);
266 }

268 PassPtr aas_routing_pass(
269  const Architecture& arc, const unsigned lookahead,
270  const aas::CNotSynthType cnotsynthtype) {
271  Transform::Transformation trans = [=](Circuit& circ) {
272  // check input:
273  if (lookahead == 0) {
274  throw std::logic_error("lookahead must be > 0");
275  }
276  if (arc.n_uids() < circ.n_qubits()) {
277  throw CircuitInvalidity(
278  "Circuit has more qubits than the architecture has nodes.");
279  }
280 
281  qubit_vector_t all_qu = circ.all_qubits();
282 
283  Circuit input_circ = circ;
284 
285  VertexList bin;
286 
287  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
288  if (!circ.detect_boundary_Op(v)) {
289  bin.push_back(v);
290  }
291  }
292 
293  circ.remove_vertices(
294  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
295 
296  for (const Command& com : input_circ) {
297  OpType ot = com.get_op_ptr()->get_type();
298  unit_vector_t qbs = com.get_args();
299  switch (ot) {
300  case OpType::PhasePolyBox: {
301  Op_ptr op = com.get_op_ptr();
302  const PhasePolyBox& b = static_cast<const PhasePolyBox&>(*op);
303 
304  Circuit b_circ(*b.to_circuit());
305 
306  PhasePolyBox ppb(b_circ);
307 
308  Circuit result =
309  aas::phase_poly_synthesis(arc, ppb, lookahead, cnotsynthtype);
310 
311  for (const Command& res_com : result) {
312  OpType ot = res_com.get_op_ptr()->get_type();
313  unit_vector_t res_qbs = res_com.get_args();
314 
315  switch (ot) {
316  case OpType::CX: {
317  circ.add_op<Node>(ot, {Node(res_qbs[0]), Node(res_qbs[1])});
318  break;
319  }
320  case OpType::Rz: {
321  auto angle = res_com.get_op_ptr()->get_params().at(0);
322  circ.add_op<Node>(ot, angle, {Node(res_qbs[0])});
323  break;
324  }
325  default: {
326  TKET_ASSERT(!"Invalid gate type in phase poly box");
327  }
328  }
329  }
330  break;
331  }
332  case OpType::H: {
333  circ.add_op<Node>(ot, {Node(qbs[0])});
334  break;
335  }
336  case OpType::Collapse: {
337  circ.add_op<Node>(ot, {Node(qbs[0])});
338  break;
339  }
340  case OpType::Reset: {
341  circ.add_op<Node>(ot, {Node(qbs[0])});
342  break;
343  }
344  case OpType::Measure: {
345  circ.add_measure(Node(qbs[0]), Bit(qbs[1]));
346  break;
347  }
348  default: {
349  throw NotImplemented(
350  "Invalid gate in input of aas routing: " +
351  com.get_op_ptr()->get_name());
352  }
353  }
354  }
355 
356  return true;
357  };
358  Transform t = Transform(trans);
359 
360  PredicatePtr placedpred = std::make_shared<PlacementPredicate>(arc);
361  PredicatePtr n_qubit_pred =
362  std::make_shared<MaxNQubitsPredicate>(arc.n_uids());
363  PredicatePtrMap precons{
364  CompilationUnit::make_type_pair(placedpred),
365  CompilationUnit::make_type_pair(n_qubit_pred)};
366 
367  PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
368  std::pair<const std::type_index, PredicatePtr> pair1 =
369  CompilationUnit::make_type_pair(postcon1);
370  PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
371  PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
372  std::type_index gateset_ti = typeid(GateSetPredicate);
373  PredicateClassGuarantees g_postcons{
374  {pair1.first, Guarantee::Clear}, {gateset_ti, Guarantee::Clear}};
375  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
376 
377  nlohmann::json j;
378  j["name"] = "AASRoutingPass";
379  j["architecture"] = arc;
380 
381  return std::make_shared<StandardPass>(precons, t, pc, j);
382 }

402 PassPtr gen_decompose_routing_gates_to_cxs_pass(
403  const Architecture& arc, bool directed) {
404  PredicateClassGuarantees g_postcons{
405  {typeid(GateSetPredicate), Guarantee::Clear}};
406  PredicatePtrMap precons;
407  PredicatePtrMap s_postcons;
408  Transform t = Transform::decompose_SWAP_to_CX(arc) >>
409  Transform::decompose_BRIDGE_to_CX() >>
410  Transform::remove_redundancies();
411  if (directed) {
412  OpTypeSet out_optypes{all_single_qubit_types()};
413  out_optypes.insert(OpType::CX);
414  OpTypeSet in_optypes = out_optypes;
415  in_optypes.insert(OpType::SWAP);
416  in_optypes.insert(OpType::BRIDGE);
417  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
418  PredicatePtr connected = std::make_shared<ConnectivityPredicate>(arc);
419  PredicatePtr wireswaps = std::make_shared<NoWireSwapsPredicate>();
420  PredicatePtr directedpred = std::make_shared<DirectednessPredicate>(arc);
421  PredicatePtr ingates = std::make_shared<GateSetPredicate>(in_optypes);
422  PredicatePtr outgates = std::make_shared<GateSetPredicate>(out_optypes);
423  precons = {
424  CompilationUnit::make_type_pair(connected),
425  CompilationUnit::make_type_pair(wireswaps),
426  CompilationUnit::make_type_pair(ingates)};
427  s_postcons = {
428  CompilationUnit::make_type_pair(directedpred),
429  CompilationUnit::make_type_pair(outgates),
430  CompilationUnit::make_type_pair(twoqbpred)};
431  t = t >> Transform::decompose_CX_directed(arc) >>
432  Transform::remove_redundancies();
433  }
434  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
435  // record pass config
436  nlohmann::json j;
437  j["name"] = "DecomposeSwapsToCXs";
438  j["directed"] = directed;
439  j["architecture"] = arc;
440  return std::make_shared<StandardPass>(precons, t, pc, j);
441 }
