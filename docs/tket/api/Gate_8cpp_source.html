44  return get_op_ptr(optype);
45  }
46  case OpType::CnX: {
47  return get_op_ptr(optype, std::vector<Expr>(), n_qubits_);
48  }
49  case OpType::S: {
50  return get_op_ptr(OpType::Sdg);
51  }
52  case OpType::Sdg: {
53  return get_op_ptr(OpType::S);
54  }
55  case OpType::T: {
56  return get_op_ptr(OpType::Tdg);
57  }
58  case OpType::Tdg: {
59  return get_op_ptr(OpType::T);
60  }
61  case OpType::V: {
62  return get_op_ptr(OpType::Vdg);
63  }
64  case OpType::Vdg: {
65  return get_op_ptr(OpType::V);
66  }
67  case OpType::CV: {
68  return get_op_ptr(OpType::CVdg);
69  }
70  case OpType::CVdg: {
71  return get_op_ptr(OpType::CV);
72  }
73  case OpType::SX: {
74  return get_op_ptr(OpType::SXdg);
75  }
76  case OpType::SXdg: {
77  return get_op_ptr(OpType::SX);
78  }
79  case OpType::CSX: {
80  return get_op_ptr(OpType::CSXdg);
81  }
82  case OpType::CSXdg: {
83  return get_op_ptr(OpType::CSX);
84  }
85  case OpType::CRz:
86  case OpType::CRx:
87  case OpType::CRy:
88  case OpType::CU1:
89  case OpType::U1:
90  case OpType::Rz:
91  case OpType::Ry:
92  case OpType::Rx:
93  case OpType::PhaseGadget:
94  case OpType::CnRy:
95  case OpType::XXPhase:
96  case OpType::YYPhase:
97  case OpType::ZZPhase:
98  case OpType::XXPhase3:
99  case OpType::ISWAP:
100  case OpType::ESWAP: {
101  return get_op_ptr(optype, -params_[0], n_qubits_);
102  }
103  case OpType::ZZMax: {
104  // ZZMax.dagger = ZZPhase(-0.5)
105  return get_op_ptr(OpType::ZZPhase, -0.5);
106  }
107  case OpType::FSim: {
108  // FSim(a,b).dagger() == FSim(-a,-b)
109  return get_op_ptr(OpType::FSim, {-params_[0], -params_[1]});
110  }
111  case OpType::Sycamore: {
112  return get_op_ptr(OpType::FSim, {-0.5, -1. / 6.});
113  }
114  case OpType::ISWAPMax: {
115  return get_op_ptr(OpType::ISWAP, 3.);
116  }
117  case OpType::U2: {
118  // U2(a,b).dagger() == U3(-pi/2,-b,-a)
119  return get_op_ptr(OpType::U3, {-0.5, -params_[1], -params_[0]});
120  }
121  case OpType::U3:
122  case OpType::CU3:
123  // U3(a,b,c).dagger() == U3(-a,-c.-b)
124  { return get_op_ptr(optype, {-params_[0], -params_[2], -params_[1]}); }
125  case OpType::TK1:
126  // TK1(a,b,c).dagger() == TK1(-c,-b,-a)
127  {
128  return get_op_ptr(OpType::TK1, {-params_[2], -params_[1], -params_[0]});
129  }
130  case OpType::PhasedX:
131  case OpType::NPhasedX:
132  // PhasedX(a,b).dagger() == PhasedX(-a,b)
133  { return get_op_ptr(optype, {-params_[0], params_[1]}, n_qubits_); }
134  case OpType::PhasedISWAP:
135  // PhasedISWAP(a,b).dagger() == PhasedISWAP(a,-b)
136  { return get_op_ptr(OpType::PhasedISWAP, {params_[0], -params_[1]}); }
137  default: {
138  throw NotImplemented(
139  "Cannot retrieve the dagger of OpType::" + get_desc().name());
140  }

196  return get_op_ptr(OpType::U2, {params_[1] + 1., params_[0] + 1.});
197  }

199  case OpType::CU3: {

208  case OpType::NPhasedX: {

214  return get_op_ptr(OpType::PhasedISWAP, {-params_[0], params_[1]});
215  }
216 
217  default: {
218  throw NotImplemented(

233 std::optional<double> Gate::is_identity() const {
234  static const std::optional<double> notid;
235  const std::vector<Expr>& params = get_params();
236  switch (get_type()) {
237  case OpType::Rx:
238  case OpType::Ry:
239  case OpType::Rz:
240  case OpType::PhasedX:
241  case OpType::NPhasedX:
242  case OpType::XXPhase:
243  case OpType::YYPhase:
244  case OpType::ZZPhase:
245  case OpType::XXPhase3:
246  case OpType::ESWAP: {
247  Expr e = params[0];
248  if (equiv_0(e, 4)) {
249  return 0.;
250  } else if (equiv_0(e + 2, 4)) {
251  return 1.;
252  } else
253  return notid;
254  }
255  case OpType::U1:
256  case OpType::CU1: {
257  return equiv_0(params[0]) ? 0. : notid;
258  }
259  case OpType::U3: {
260  Expr theta = params[0];
261  if (equiv_0(params[1] + params[2])) {
262  if (equiv_0(theta, 4)) {
263  return 0.;
264  } else if (equiv_0(theta + 2, 4)) {
265  return 1.;
266  } else
267  return notid;
268  } else
269  return notid;
270  }
271  case OpType::CU3: {
272  if (equiv_0(params[0], 4) && equiv_0(params[1] + params[2])) {
273  return 0.;
274  } else
275  return notid;
276  }
277  case OpType::TK1: {
278  Expr s = params[0] + params[2], t = params[1];
279  if (equiv_0(s) && equiv_0(t)) {
280  return (equiv_0(s, 4) ^ equiv_0(t, 4)) ? 1. : 0.;
281  } else
282  return notid;
283  }
284  case OpType::CRz:
285  case OpType::CRx:
286  case OpType::CRy:
287  case OpType::PhaseGadget:
288  case OpType::ISWAP:
289  case OpType::CnRy: {
290  return equiv_0(params[0], 4) ? 0. : notid;
291  }
292  case OpType::FSim: {
293  return (equiv_0(params[0]) && equiv_0(params[1])) ? 0. : notid;
294  }
295  case OpType::PhasedISWAP: {
296  return (equiv_0(params[0], 1) && equiv_0(params[1], 4)) ? 0. : notid;
297  }
298  default:
299  return notid;
300  }
301 }

303 std::string Gate::get_name(bool latex) const {
304  OpDesc desc = get_desc();
305  if (params_.size() > 0) {
306  std::stringstream name;
307  if (latex) {
308  name << desc.latex() << "(";
309  } else {
310  name << desc.name() << "(";
311  }
312  for (unsigned i = 0; i < params_.size(); ++i) {
313  std::optional<double> reduced =
314  eval_expr_mod(params_[i], desc.param_mod(i));
315  if (reduced) {
316  name << reduced.value();
317  } else {
318  name << params_[i];
319  }
320  if (i < params_.size() - 1) name << ", ";
321  }
322  name << ")";
323  return name.str();
324  } else {
325  return Op::get_name(latex);
326  }

348 bool Gate::is_equal(const Op& op_other) const {
349  const Gate& other = dynamic_cast<const Gate&>(op_other);
350 
351  OpDesc desc = get_desc();
352  if (n_qubits() != other.n_qubits()) return false;
353  std::vector<Expr> params1 = this->get_params();
354  std::vector<Expr> params2 = other.get_params();
355  unsigned param_count = params1.size();
356  if (params2.size() != param_count) return false;
357  for (unsigned i = 0; i < param_count; i++) {
358  if (!equiv_expr(params1[i], params2[i], desc.param_mod(i))) return false;
359  }
360  return true;
361 }

379 std::vector<Expr> Gate::get_tk1_angles() const {
380  const Expr half =
381  SymEngine::div(SymEngine::integer(1), SymEngine::integer(2));
382  const Expr quarter =
383  SymEngine::div(SymEngine::integer(1), SymEngine::integer(4));
384  const Expr eighth =
385  SymEngine::div(SymEngine::integer(1), SymEngine::integer(8));
386  switch (get_type()) {
387  case OpType::noop: {
388  return {0, 0, 0, 0};
389  }
390  case OpType::Z: {
391  return {0, 0, 1, half};
392  }
393  case OpType::X: {
394  return {0, 1, 0, half};
395  }
396  case OpType::Y: {
397  return {half, 1, -half, half};
398  }
399  case OpType::S: {
400  return {0, 0, half, quarter};
401  }
402  case OpType::Sdg: {
403  return {0, 0, -half, -quarter};
404  }
405  case OpType::T: {
406  return {0, 0, quarter, eighth};
407  }
408  case OpType::Tdg: {
409  return {0, 0, -quarter, -eighth};
410  }
411  case OpType::V: {
412  return {0, half, 0, 0};
413  }
414  case OpType::Vdg: {
415  return {0, -half, 0, 0};
416  }
417  case OpType::SX: {
418  return {0, half, 0, quarter};
419  }
420  case OpType::SXdg: {
421  return {0, -half, 0, -quarter};
422  }
423  case OpType::H: {
424  return {half, half, half, half};
425  }
426  case OpType::Rx: {
427  return {0, params_.at(0), 0, 0};
428  }
429  case OpType::Ry: {
430  return {half, params_.at(0), -half, 0};
431  }
432  case OpType::Rz:
433  case OpType::PhaseGadget: {
434  return {0, 0, params_.at(0), 0};
435  }
436  case OpType::U1: {
437  return {0, 0, params_.at(0), params_.at(0) / 2};
438  }
439  case OpType::U2: {
440  return {
441  params_.at(0) + half, half, params_.at(1) - half,
442  (params_.at(0) + params_.at(1)) / 2};
443  }
444  case OpType::U3: {
445  return {
446  params_.at(1) + half, params_.at(0), params_.at(2) - half,
447  (params_.at(1) + params_.at(2)) / 2};
448  }
449  case OpType::NPhasedX: {
450  if (n_qubits_ != 1) {
451  throw NotValid(
452  "OpType::NPhasedX can only be decomposed into a TK1 "
453  "if it acts on a single qubit");
454  }
455  return {params_.at(1), params_.at(0), -params_.at(1), 0};
456  }
457  case OpType::PhasedX: {
458  return {params_.at(1), params_.at(0), -params_.at(1), 0};
459  }
460  case OpType::TK1: {
461  return {params_.at(0), params_.at(1), params_.at(2), 0};
462  }
463  default: {
464  throw NotImplemented(
465  "Cannot retrieve the TK1 angles of OpType::" + get_desc().name());
466  }
467  }

474 std::optional<Pauli> Gate::commuting_basis(port_t port) const {
475  unsigned n_q = n_qubits();
476  if (port >= n_q) throw NotValid();
477  switch (get_type()) {
478  case OpType::XXPhase:
479  case OpType::XXPhase3: {
480  return Pauli::X;
481  }
482  case OpType::YYPhase: {
483  return Pauli::Y;
484  }
485  case OpType::Z:
486  case OpType::S:
487  case OpType::Sdg:
488  case OpType::T:
489  case OpType::Tdg:
490  case OpType::Rz:
491  case OpType::U1:
492  case OpType::CZ:
493  case OpType::CRz:
494  case OpType::CU1:
495  case OpType::PhaseGadget:
496  case OpType::ZZMax:
497  case OpType::ZZPhase: {
498  return Pauli::Z;
499  }
500  case OpType::noop: {
501  return Pauli::I;
502  }
503  case OpType::H:
504  case OpType::U3:
505  case OpType::U2:
506  case OpType::PhasedX:
507  case OpType::NPhasedX:
508  case OpType::TK1: {
509  return std::nullopt;
510  }
511  case OpType::CH:
512  case OpType::CU3:
513  case OpType::CSWAP: {
514  if (port == 0) {
515  return Pauli::Z;
516  } else {
517  return std::nullopt;
518  }
519  }
520  case OpType::BRIDGE: {
521  if (port == 0) {
522  return Pauli::Z;
523  } else if (port == 2) {
524  return Pauli::X;
525  } else {
526  return Pauli::I;
527  }
528  }
529  case OpType::X:
530  case OpType::V:
531  case OpType::Vdg:
532  case OpType::CV:
533  case OpType::CVdg:
534  case OpType::SX:
535  case OpType::SXdg:
536  case OpType::CSX:
537  case OpType::CSXdg:
538  case OpType::Rx:
539  case OpType::CRx:
540  case OpType::CX:
541  case OpType::CCX:
542  case OpType::CnX: {
543  if (port == n_q - 1) {
544  return Pauli::X;
545  } else {
546  return Pauli::Z;
547  }
548  }
549  case OpType::ECR: {
550  if (port == 1) {
551  return Pauli::X;
552  } else {
553  return std::nullopt;
554  }
555  }
556  case OpType::Y:
557  case OpType::Ry:
558  case OpType::CY:
559  case OpType::CRy:
560  case OpType::CnRy: {
561  if (port == n_q - 1) {
562  return Pauli::Y;
563  } else {
564  return Pauli::Z;
565  }
566  }
567  default: {
568  return std::nullopt;
569  }
570  }
571 }

605 Op_ptr Gate::deserialize(const nlohmann::json& j) {
606  OpType optype = j.at("type").get<OpType>();
607  std::vector<Expr> params;
608  if (j.contains("params")) {
609  params = j.at("params").get<std::vector<Expr>>();
610  }
611  // if type has fixed number of qubits use it, otherwise it should have been
612  // stored
613  const auto& sig = optypeinfo().at(optype).signature;
614  unsigned n_qb;
615  if (sig) {
616  auto check_quantum = [](unsigned sum, const EdgeType& e) {
617  return sum + (e == EdgeType::Quantum ? 1 : 0);
618  };
619  n_qb = std::accumulate(sig->begin(), sig->end(), 0, check_quantum);
620  } else {
621  n_qb = j.at("n_qb").get<unsigned>();
622  }
623  return get_op_ptr(optype, params, n_qb);
624 }
