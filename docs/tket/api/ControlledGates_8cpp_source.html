42 // `n` = size of incrementer, circuit returned is of size `n+1`
43 /* this is slightly less efficient than perhaps it could be -- asymptotically it
44  is still good. In an ideal world, this would decompose the incrementers
45  smarter for the "even" case */
46 Circuit Transform::incrementer_borrow_1_qubit(unsigned n) {
47  bool is_odd = n % 2;
48  Circuit circ(n + 1);
49  if (n < 4) {
50  if (n > 2) circ.add_op<unsigned>(OpType::CCX, {0, 1, 2});
51  if (n > 1) circ.add_op<unsigned>(OpType::CX, {0, 1});
52  if (n > 0) circ.add_op<unsigned>(OpType::X, {0});
53  return circ;
54  }
55  unsigned j;
56  unsigned k;
57  // j is bottom qubits, k is top qubits
58  // k + j = n + 1 (total no. of qbs)
59  if (is_odd) {
60  /* if the number of bits we are incrementing is odd, we can just split the
61  incrementer in 2 and use the `incrementer_borrow_n_qubits` method twice */
62  j = (n + 1) / 2, k = (n + 1) / 2;
63  } else {
64  /* otherwise, we will also have to pull out a cnx */
65  j = n / 2 + 1, k = n / 2;
66  }
67 
68  Circuit top_incrementer = Transform::incrementer_borrow_n_qubits(k);
69  std::vector<unsigned> top_qbs(2 * k);
70  for (unsigned i = 0; i != k; ++i) {
71  top_qbs[2 * i] = i + k; // borrowed qubits
72  top_qbs[2 * i + 1] = i; // qbs we are trying to increment
73  }
74 
75  Circuit cnx_top;
76  std::vector<unsigned> cnx1_qbs;
77  if (k == 2) { // code is unreachable if k<2
78  cnx_top = CircPool::CCX_normal_decomp();
79  cnx1_qbs = {0, 1, n};
80  } else {
81  cnx_top = lemma72(k); // k controls on cnx
82  cnx1_qbs.resize(
83  2 * k - 2); // size of replacement using borrowed qbs = 2*k-1
84  std::iota(cnx1_qbs.begin(), cnx1_qbs.end(), 0);
85  cnx1_qbs.push_back(n); // target is last qubit
86  }
87 
88  Circuit bottom_incrementer;
89  std::vector<unsigned> bot_qbs;
90  if (is_odd) {
91  bottom_incrementer = Transform::incrementer_borrow_n_qubits(j);
92  bot_qbs.resize(2 * j);
93  for (unsigned i = 0; i != j; ++i) {
94  bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
95  if (i != 0)
96  bot_qbs[2 * i + 1] =
97  i + j -
98  1; // 3,5...n //other qbs we are actually trying to increment
99  }
100  bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
101  } else {
102  if (j == 3) { // code is unreachable if j<3
103  bottom_incrementer.add_blank_wires(3);
104  bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
105  bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
106  bottom_incrementer.add_op<unsigned>(OpType::X, {0});
107  bot_qbs = {n, n - 2, n - 1};
108  } else {
109  // insert peeled-out cnx
110  Circuit cnx_bot = lemma72(j - 1);
111  std::vector<unsigned> cnx2_qbs(
112  2 * j - 3); // lemma 7.2 uses 2j-3 qubits for a (j-1)-controlled X
113  for (unsigned i = 0; i < j - 2; ++i) {
114  cnx2_qbs[i] = k + i;
115  }
116  cnx2_qbs[j - 2] = n; // replace the last control
117  for (unsigned i = 0; i < j - 3; ++i) {
118  cnx2_qbs[j + i - 1] = i;
119  }
120  cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
121 
122  circ.append_qubits(cnx_bot, cnx2_qbs);
123  bottom_incrementer = Transform::incrementer_borrow_n_qubits(
124  j - 1); // insert incrementer over remaining qubits
125  bot_qbs.resize(2 * j - 2);
126  for (unsigned i = 0; i != j - 1; ++i) {
127  bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
128  if (i != 0)
129  bot_qbs[2 * i + 1] =
130  i + k -
131  1; // 3,5...n //other qbs we are actually trying to increment
132  }
133  bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
134  }
135  }
136 
137  circ.append_qubits(bottom_incrementer, bot_qbs);
138  circ.add_op<unsigned>(
139  OpType::X,
140  {n}); // to convert controlled-incrementer to larger incrementer
141  for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
142  circ.append_qubits(cnx_top, cnx1_qbs);
143  if (!is_odd && j > 3) {
144  // insert peeled-out cnx
145  Circuit cnx_bot = lemma72(j - 1);
146  std::vector<unsigned> cnx2_qbs(2 * j - 3);
147  for (unsigned i = 0; i < j - 1; ++i) {
148  cnx2_qbs[i] = k + i;
149  }
150  cnx2_qbs[j - 2] = n; // the last control
151  for (unsigned i = 0; i < j - 3; ++i) {
152  cnx2_qbs[j + i - 1] = i;
153  }
154  cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
155  circ.append_qubits(cnx_bot, cnx2_qbs);
156  }
157  circ.append_qubits(bottom_incrementer, bot_qbs);
158  circ.add_op<unsigned>(OpType::X, {n});
159  circ.append_qubits(cnx_top, cnx1_qbs);
160  for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
161  circ.append_qubits(top_incrementer, top_qbs);
162  return circ;
163 }

165 // an optimised version of
166 // https://algassert.com/circuits/2015/06/12/Constructing-Large-Increment-Gates.html
167 /* every second qubit (0,2,4...) is a borrowed qubit */
168 Circuit Transform::incrementer_borrow_n_qubits(unsigned n) {
169  const unsigned N = 2 * n;
170  Circuit circ(N);
171  /* deal with small cases where borrowing qubits is unnecessary */
172  if (n < 4) {
173  if (n > 2) circ.add_op<unsigned>(OpType::CCX, {1, 3, 5});
174  if (n > 1) circ.add_op<unsigned>(OpType::CX, {1, 3});
175  if (n > 0) circ.add_op<unsigned>(OpType::X, {1});
176  return circ;
177  }
178 
179  for (unsigned i = 1; i < N; ++i) {
180  if (i % 2)
181  circ.add_op<unsigned>(OpType::CX, {0, i});
182  else
183  circ.add_op<unsigned>(OpType::X, {i});
184  }
185 
186  circ.add_op<unsigned>(OpType::X, {N - 1});
187 
188  for (unsigned i = 2; i < N; ++(++i)) {
189  std::vector<unsigned> ladder_down_qbs = {i - 2, i - 1, i};
190  circ.append_qubits(CircPool::ladder_down(), ladder_down_qbs);
191  }
192  circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
193  for (unsigned i = N - 2; i > 1; --(--i)) {
194  std::vector<unsigned> tof_up_qbs = {i - 2, i - 1, i};
195  circ.add_op<unsigned>(OpType::CCX, tof_up_qbs);
196  }
197 
198  for (unsigned i = 2; i < N; ++(++i)) {
199  std::vector<unsigned> ladder_down_2_qbs = {i - 2, i - 1, i};
200  circ.append_qubits(CircPool::ladder_down_2(), ladder_down_2_qbs);
201  }
202  circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
203  for (unsigned i = N - 2; i > 1; --(--i)) {
204  std::vector<unsigned> ladder_up_qbs = {i - 2, i - 1, i};
205  circ.append_qubits(CircPool::ladder_up(), ladder_up_qbs);
206  }
207  for (unsigned i = 1; i < N; ++(++i))
208  circ.add_op<unsigned>(OpType::CX, {0, i});
209  return circ;
210 }

212 // decompose CnX gate using
213 // https://algassert.com/circuits/2015/06/22/Using-Quantum-Gates-instead-of-Ancilla-Bits.html
214 // `n` = no. of controls
215 Circuit Transform::cnx_normal_decomp(unsigned n) {
216  /* handle low qb edge cases */
217  bool insert_ccxs; // dictate whether to add Toffolis or n>3 controlled Xs
218  // when bootstrapping
219  switch (n) {
220  case 0: {
221  return CircPool::X();
222  }
223  case 1: {
224  return CircPool::CX();
225  }
226  case 2: {
227  return CircPool::CCX_normal_decomp();
228  }
229  case 3: {
230  insert_ccxs = true;
231  break;
232  }
233  default: {
234  insert_ccxs = false;
235  break;
236  }
237  }
238 
239  Circuit circ(n + 1);
240  std::vector<unsigned> cnx_qbs(n - 1);
241  std::iota(cnx_qbs.begin(), cnx_qbs.end(), 0);
242  cnx_qbs.push_back(n);
243  // first, bootstrap an ancilla qubit
244  circ.add_op<unsigned>(OpType::H, {n});
245  Vertex cnx1;
246  if (insert_ccxs)
247  cnx1 = circ.add_op<unsigned>(OpType::CCX, {cnx_qbs});
248  else {
249  cnx1 = circ.add_op<unsigned>(
250  OpType::CnX,
251  {cnx_qbs}); /* the CnXs can be decomposed using lemma 7.3 */
252  }
253  circ.add_op<unsigned>(OpType::Tdg, {n});
254  Vertex cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
255  if (!insert_ccxs) {
256  Edge e1 = circ.get_nth_in_edge(cx, 0);
257  lemma73(circ, {e1, cnx1}); // replace first CnX using lemma 7.3
258  }
259  circ.add_op<unsigned>(OpType::T, {n});
260  Vertex cnx2;
261  if (insert_ccxs)
262  cnx2 = circ.add_op<unsigned>(OpType::CCX, {cnx_qbs});
263  else {
264  cnx2 = circ.add_op<unsigned>(OpType::CnX, {cnx_qbs});
265  }
266  circ.add_op<unsigned>(OpType::Tdg, {n});
267  cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
268  Edge e2 = circ.get_nth_in_edge(cx, 0);
269  if (!insert_ccxs) lemma73(circ, {e2, cnx2});
270  circ.add_op<unsigned>(OpType::T, {n});
271  circ.add_op<unsigned>(OpType::H, {n});
272 
273  // add incremented shift pattern
274  Circuit incrementer = incrementer_borrow_1_qubit(n);
275  circ.append(incrementer);
276 
277  // z rotation layer #1
278  std::vector<Op_ptr> z_rots(n);
279  double angle = -0.25;
280  for (unsigned i = 0; i < n - 1; ++i) {
281  unsigned m = n - 1 - i;
282  z_rots[i] = get_op_ptr(OpType::Rz, angle);
283  circ.add_op<unsigned>(z_rots[i], {m});
284  angle /= 2;
285  }
286 
287  // decremented shift pattern
288  for (unsigned i = 0; i < n; ++i) {
289  circ.add_op<unsigned>(OpType::X, {i});
290  }
291  circ.append(incrementer);
292  for (unsigned i = 0; i < n; ++i) {
293  circ.add_op<unsigned>(OpType::X, {i});
294  }
295 
296  // z rotation layer #2
297  for (unsigned i = 0; i < n - 1; ++i) {
298  unsigned m = n - 1 - i;
299  Expr ang = z_rots[i]->get_params()[0];
300  circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {m});
301  }
302  Expr ang = z_rots[n - 2]->get_params()[0];
303  circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {0});
304 
305  Transform::decomp_CCX().apply(circ);
306  circ.add_phase(std::pow(0.5, n + 1));
307  return circ;
308 }

310 /* assumes vert is controlled Ry with 1 control */
311 /* decomposes CRy into 2 CXs and 2 Ry gates */
312 static Circuit lemma54(const Expr& angle) {
313  Circuit new_circ(2);
314  std::vector<Expr> A_params = {angle / 2.};
315  std::vector<Expr> B_params = {-angle / 2.};
316  const Op_ptr A = get_op_ptr(OpType::Ry, A_params);
317  const Op_ptr B = get_op_ptr(OpType::Ry, B_params);
318  new_circ.add_op<unsigned>(A, {1});
319  new_circ.add_op<unsigned>(OpType::CX, {0, 1});
320  new_circ.add_op<unsigned>(B, {1});
321  new_circ.add_op<unsigned>(OpType::CX, {0, 1});
322  return new_circ;
323 }

325 // unsigned -> which column the change is in
326 static unsigned find_first_differing_val(
327  const std::deque<bool>& d1, const std::deque<bool>& d2) {
328  unsigned N = d1.size();
329  if (N != d2.size())
330  throw ControlDecompError(
331  "Error in `find_first_differing_val`: Deques are of differing "
332  "sizes");
333  for (unsigned i = 0; i < N; ++i) {
334  if (d1[i] != d2[i]) return i;
335  }
336  throw ControlDecompError(
337  "Error in `find_first_differing_val`: No change between deques");
338 }

340 // optimal decomposition of CnRy for 2 < n < 8 according to 1995 paper... can do
341 // better with ZH calculus?
342 static Circuit lemma71(unsigned arity, const Expr& angle) {
343  unsigned m_controls = arity - 1;
344  if (m_controls < 2)
345  throw Unsupported(
346  "No point using Lemma 7.1 to decompose a gate with less than 2 "
347  "controls");
348  if (m_controls > 7)
349  throw Unsupported(
350  "Using Lemma 7.1 to decompose a gate with more than 7 controls "
351  "is inefficient");
352 
353  GrayCode gc = gen_graycode(m_controls);
354  unsigned n_square_roots = m_controls - 1;
355 
356  Circuit rep(arity);
357  Expr param;
358  std::optional<double> reduced = eval_expr_mod(angle, 4);
359  if (reduced)
360  param = reduced.value();
361  else
362  param = angle;
363 
364  param = param / (1 << (n_square_roots));
365 
366  const Op_ptr V_op = get_op_ptr(OpType::CnRy, param, 2);
367  const Op_ptr V_dg = get_op_ptr(OpType::CnRy, -param, 2);
368 
369  unsigned control_qb = 0;
370  unsigned last = 0;
371  rep.add_op<unsigned>(V_op, {0, m_controls});
372  // we ignore the 0...0 term, and the first one is always trivial
373  // so start from 2
374  for (unsigned i = 2; i < gc.size(); ++i) {
375  unsigned change = find_first_differing_val(gc[i], gc[i - 1]);
376  for (unsigned j = 1; j < gc[i].size(); ++j) {
377  if (gc[i][j] == 1) last = j;
378  }
379  if (change < control_qb) {
380  rep.add_op<unsigned>(OpType::CX, {change, control_qb});
381  }
382  if (change == control_qb) {
383  throw ControlDecompError("Error in graycode iteration");
384  }
385  if (change > control_qb) {
386  rep.add_op<unsigned>(OpType::CX, {control_qb, change});
387  }
388 
389  if ((i % 2) == 0) {
390  rep.add_op<unsigned>(V_dg, {last, m_controls});
391  } else
392  rep.add_op<unsigned>(V_op, {last, m_controls});
393  control_qb = last;
394  }
395  unsigned correct_gate_count =
396  ((1 << m_controls) - 1) + ((1 << m_controls) - 2);
397  if (rep.n_gates() != correct_gate_count)
398  throw ControlDecompError("Error in Lemma 7.1: Gate count is incorrect");
399  auto [vit, vend] = boost::vertices(rep.dag);
400  for (auto next = vit; vit != vend; vit = next) {
401  ++next;
402  Vertex v = *vit;
403  if (rep.get_OpType_from_Vertex(v) == OpType::CnRy) {
404  Expr v_angle = rep.get_Op_ptr_from_Vertex(v)->get_params()[0];
405  Circuit cnry_replacement = lemma54(v_angle);
406  Subcircuit sub{rep.get_in_edges(v), rep.get_all_out_edges(v), {v}};
407  rep.substitute(cnry_replacement, sub, Circuit::VertexDeletion::Yes);
408  }
409  }
410  return rep;
411 }

413 //`control_m` = number of controls
414 static Circuit lemma72(unsigned control_m) {
415  if (control_m < 3)
416  throw Unsupported(
417  "Cannot decompose a gate with " + std::to_string(control_m) +
418  " controls using Lemma 7.2");
419  const unsigned n = 2 * control_m - 1;
420 
421  Circuit ccx_circ(n);
422  unsigned diff = n - control_m;
423  for (unsigned i = control_m - 1; i > 1; --i) {
424  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
425  }
426  ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
427  for (unsigned i = 2; i < control_m; ++i) {
428  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
429  }
430  for (unsigned i = control_m - 2; i > 1; --i) {
431  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
432  }
433  ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
434  for (unsigned i = 2; i < control_m - 1; ++i) {
435  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
436  }
437  if (ccx_circ.count_gates(OpType::CCX) != 4 * (control_m - 2))
438  throw ControlDecompError("Error in Lemma 7.2: CCX gate count is incorrect");
439  return ccx_circ;
440 }

442 /* this is specifically for performing corollary 7.4 via lemma 7.3 & lemma 7.2 -
443  * optimal decomp of a CnX gate */
444 /* for corollary 7.4, n >= 7 */
445 // this is a decomposition of a CnX gate using one dirty ancilla
446 static void lemma73(Circuit& circ, const std::pair<Edge, Vertex>& pairy) {
447  Vertex original_cnx = pairy.second;
448  Edge original_spare_edge = pairy.first;
449  EdgeVec in_edges = circ.get_in_edges(original_cnx);
450  const unsigned N =
451  in_edges.size() + 1; // number of qubits in new circuit to replace
452  if (N < 5)
453  throw Unsupported(
454  "Lemma 7.3 cannot decompose CnX with n = " + std::to_string(N - 1));
455 
456  EdgeVec out_edges = circ.get_all_out_edges(original_cnx);
457 
458  in_edges.insert(in_edges.begin() + in_edges.size() - 1, original_spare_edge);
459  out_edges.insert(
460  out_edges.begin() + out_edges.size() - 1, original_spare_edge);
461 
462  Subcircuit to_delete{in_edges, out_edges, {original_cnx}};
463  bool odd_N = N % 2;
464  unsigned m1 = (N + 1) / 2; // number of controls on the first type of CnX
465  unsigned m2 = N - m1 - 1; // number of controls on the second type of CnX
466 
467  // make new circuit to substitute later
468  Circuit new_circ(N);
469  const Op_ptr cnx_op1 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m1 + 1);
470  const Op_ptr cnx_op2 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m2 + 1);
471  std::vector<unsigned> qbs_m1(m1 + 1);
472  std::iota(qbs_m1.begin(), qbs_m1.begin() + m1, 0);
473  qbs_m1[m1] = N - 1;
474  std::vector<unsigned> qbs_m2(m2 + 1);
475  std::iota(qbs_m2.begin(), qbs_m2.end(), N - (m2 + 1));
476 
477  // add ladder of CnXs to the correct qubits
478  Vertex a = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
479  Vertex b = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
480  Vertex c = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
481  Vertex d = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
482 
483  /* first, replace vertex a, putting circuit at back of new_circ */
484  unsigned cutsize = odd_N ? N : N - 1;
485  EdgeVec cut1(cutsize);
486  VertexVec out_verts = new_circ.q_outputs();
487  if (odd_N) {
488  for (unsigned i = 0; i < N - 2; ++i)
489  cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
490  cut1[N - 2] = new_circ.get_nth_in_edge(out_verts[N - 1], 0);
491  cut1[N - 1] = new_circ.get_nth_in_edge(out_verts[N - 2], 0);
492  } else {
493  for (unsigned i = 0; i < cutsize; ++i)
494  cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
495  }
496 
497  Circuit a_replacement;
498  if (m1 == 1) {
499  a_replacement = CircPool::CX();
500  } else if (m1 == 2) {
501  a_replacement = CircPool::CCX();
502  } else
503  a_replacement = lemma72(m1); // returns circuit of size 2*m1 - 1
504  new_circ.cut_insert(a_replacement, cut1);
505  new_circ.remove_vertex(
506  a, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
507 
508  VertexSet normal_decomp_vertices;
509 
510  Circuit b_replacement;
511  if (m2 == 1) {
512  b_replacement = CircPool::CX();
513  } else if (m2 == 2) {
514  b_replacement = CircPool::CCX();
515  } else
516  b_replacement = lemma72(m2); // returns circuit of size 2*m2 - 1
517 
518  unsigned b_qubits = b_replacement.n_qubits();
519 
520  // reassign cut to back of circuit
521  EdgeVec frontier(N);
522  for (unsigned i = 0; i < N; ++i)

624  Circuit& replacement, unsigned N, const Expr& angle,
625  candidate_t& CCX_candidates) {
626  replacement.add_blank_wires(N);
627 
628  std::vector<Expr> A_params = {angle / 2.};
629  std::vector<Expr> B_params = {-angle / 2.};

631  const Op_ptr B = get_op_ptr(OpType::CnRy, B_params, 2);
632 
633  Vertex vA = replacement.add_op<unsigned>(A, {N - 2, N - 1}); // A
634  std::vector<unsigned> cnx_qbs(N - 1);
635  std::iota(cnx_qbs.begin(), --cnx_qbs.end(), 0);
636  cnx_qbs[N - 2] = N - 1;
637  const Op_ptr cnx = get_op_ptr(OpType::CnX, std::vector<Expr>{}, N - 1);
638  Vertex firstCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
639  Vertex vB = replacement.add_op<unsigned>(B, {N - 2, N - 1}); // B
640  CCX_candidates.push_back(
641  {boost::edge(vA, vB, replacement.dag).first, firstCnX});
642  Vertex secondCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
643  Edge out_edge_spare = replacement.get_nth_out_edge(vB, 0);
644  CCX_candidates.push_back({out_edge_spare, secondCnX});
645 }
