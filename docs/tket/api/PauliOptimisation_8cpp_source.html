22 Transform Transform::pairwise_pauli_gadgets(CXConfigType cx_config) {
23  return Transform([=](Circuit &circ) {
24  Expr t = circ.get_phase();
25  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
26  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
27  OpType optype = op->get_type();
28  if (is_boundary_c_type(optype)) continue;
29  if (optype == OpType::Conditional)
30  throw CircuitInvalidity(
31  "Cannot currently do `pauli_gadgets` optimisation on a "
32  "circuit with conditional gates");
33  if (op->get_desc().is_box())
34  throw CircuitInvalidity(
35  "Cannot currently do `pauli_gadgets` optimisation on a "
36  "circuit with boxes");
37  if (optype == OpType::Measure || optype == OpType::Collapse) {
38  VertexVec q_suc = circ.get_successors_of_type(v, EdgeType::Quantum);
39  if (q_suc.size() != 1 ||
40  !is_final_q_type(circ.get_OpType_from_Vertex(q_suc[0])))
41  throw CircuitInvalidity(
42  "Cannot currently do `pauli_gadgets` optimisation "
43  "on a circuit with a measure in the middle of the "
44  "circuit");
45  }
46  }
47  Transform setup = decompose_multi_qubits_CX() >> decompose_ZX() >>
48  decompose_ZX_to_cliffords();
49  setup.apply(circ);
50  // We effectively commute non-Clifford rotations to the front of the circuit
51  // This gives a sequence of just Pauli gadgets (gadget_circ), followed by
52  // all of the Clifford operations (clifford_circ)
53  std::vector<std::pair<QubitPauliTensor, Expr>> pauli_gadgets;
54  // rx_pauli[i] specifies which Pauli gadget would be built by applying an Rx
55  // rotation on qubit i and then pushing it through the Cliffords to the
56  // front of the circuit. Likewise for rz_pauli with Rz rotations. Clifford
57  // operations will update these and non-Clifford rotations will introduce
58  // Pauli gadgets accordingly
59  Circuit gadget_circ;
60  Circuit clifford_circ;
61  std::map<Qubit, QubitPauliTensor> rx_pauli;
62  std::map<Qubit, QubitPauliTensor> rz_pauli;
63  for (const Qubit &qb : circ.all_qubits()) {
64  gadget_circ.add_qubit(qb);
65  clifford_circ.add_qubit(qb);
66  rx_pauli.insert({qb, QubitPauliTensor(qb, Pauli::X)});
67  rz_pauli.insert({qb, QubitPauliTensor(qb, Pauli::Z)});
68  }
69  for (const Bit &cb : circ.all_bits()) {
70  gadget_circ.add_bit(cb);
71  clifford_circ.add_bit(cb);
72  }
73  // Identify Pauli Gadgets and build Clifford circuit
74  for (const Command &c : circ) {
75  const Op_ptr op_ptr = c.get_op_ptr();
76  unit_vector_t args = c.get_args();
77  OpType type = op_ptr->get_type();
78  switch (type) {
79  // Update rx_pauli and rz_pauli
80  case OpType::S: {
81  Qubit q(args[0]);
82  rx_pauli[q] = i_ * rz_pauli[q] * rx_pauli[q];
83  break;
84  }
85  case OpType::V: {
86  Qubit q(args[0]);
87  rz_pauli[q] = i_ * rx_pauli[q] * rz_pauli[q];
88  break;
89  }
90  case OpType::Z: {
91  Qubit q(args[0]);
92  rx_pauli[q] = -1. * rx_pauli[q];
93  break;
94  }
95  case OpType::X: {
96  Qubit q(args[0]);
97  rz_pauli[q] = -1. * rz_pauli[q];
98  break;
99  }
100  case OpType::Sdg: {
101  Qubit q(args[0]);
102  rx_pauli[q] = -i_ * rz_pauli[q] * rx_pauli[q];
103  break;
104  }
105  case OpType::Vdg: {
106  Qubit q(args[0]);
107  rz_pauli[q] = -i_ * rx_pauli[q] * rz_pauli[q];
108  break;
109  }
110  case OpType::CX: {
111  Qubit q_ctrl(args[0]);
112  Qubit q_trgt(args[1]);
113  rx_pauli[q_ctrl] = rx_pauli[q_ctrl] * rx_pauli[q_trgt];
114  rz_pauli[q_trgt] = rz_pauli[q_ctrl] * rz_pauli[q_trgt];
115  break;
116  }
117  // Introduce a Pauli gadget
118  case OpType::Rz: {
119  Qubit q(args[0]);
120  Expr angle = (op_ptr)->get_params()[0];
121  pauli_gadgets.push_back({rz_pauli[q], angle});
122  break;
123  }
124  case OpType::Rx: {
125  Qubit q(args[0]);
126  Expr angle = (op_ptr)->get_params()[0];
127  pauli_gadgets.push_back({rx_pauli[q], angle});
128  break;
129  }
130  case OpType::Measure:
131  case OpType::Collapse:
132  case OpType::Reset:
133  break;
134  default: {
135  std::string error_gate =
136  "Cannot perform pairwise Pauli gadget optimisation "
137  "using: " +
138  op_ptr->get_name();
139  throw NotValid(error_gate);
140  }
141  }
142  // Add Clifford gates to the back of the circuit to recreate the final
143  // combination at the outputs
144  switch (type) {
145  case OpType::Rz:
146  case OpType::Rx: {
147  break;
148  }
149  default: {
150  clifford_circ.add_op(op_ptr, args);
151  }
152  }
153  }
154  // Synthesise pairs of Pauli Gadgets
155  unsigned g = 0;
156  while (g + 1 < pauli_gadgets.size()) {
157  auto [pauli0, angle0] = pauli_gadgets[g];
158  auto [pauli1, angle1] = pauli_gadgets[g + 1];
159  append_pauli_gadget_pair(
160  gadget_circ, pauli0, angle0, pauli1, angle1, cx_config);
161  g += 2;
162  }
163  // As we synthesised Pauli gadgets 2 at a time, if there were an odd
164  // number, we will have one left over, so add that one on its own
165  if (g < pauli_gadgets.size()) {
166  auto [pauli, angle] = pauli_gadgets[g];
167  append_single_pauli_gadget(gadget_circ, pauli, angle, cx_config);
168  }
169  // Stitch gadget circuit and Clifford circuit together
170  circ = gadget_circ >> clifford_circ;
171  circ.add_phase(t);
172  clifford_simp().apply(circ);
173  return true;
174  });
175 }

177 Transform Transform::synthesise_pauli_graph(
178  PauliSynthStrat strat, CXConfigType cx_config) {
179  return Transform([=](Circuit &circ) {
180  Expr t = circ.get_phase();
181  PauliGraph pg = circuit_to_pauli_graph(circ);
182  switch (strat) {
183  case PauliSynthStrat::Individual: {
184  circ = pauli_graph_to_circuit_individually(pg, cx_config);
185  break;
186  }
187  case PauliSynthStrat::Pairwise: {
188  circ = pauli_graph_to_circuit_pairwise(pg, cx_config);
189  break;
190  }
191  case PauliSynthStrat::Sets: {
192  circ = pauli_graph_to_circuit_sets(pg, cx_config);
193  break;
194  }
195  default:
196  throw NotImplemented("Unknown Pauli Synthesis Strategy");
197  }
198  circ.add_phase(t);
199  // always turn circuit into PauliGraph and back, so always return true
200  return true;
201  });
202 }

204 Transform Transform::special_UCC_synthesis(
205  PauliSynthStrat strat, CXConfigType cx_config) {
206  return Transform([=](Circuit &circ) {
207  Transform synther = Transform::synthesise_pauli_graph(strat, cx_config);
208  // make list so we don't run into unboxing vertex issues
209  std::list<Vertex> circbox_verts;
210  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
211  if (circ.get_OpType_from_Vertex(v) == OpType::CircBox)
212  circbox_verts.push_back(v);
213  }
214  for (Vertex v : circbox_verts) {
215  const Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
216  std::shared_ptr<const CircBox> box_ptr =
217  std::dynamic_pointer_cast<const CircBox>(op);
218  Circuit inner_circ = *(box_ptr->to_circuit());
219  synther.apply(inner_circ);
220  Subcircuit sub = {circ.get_in_edges(v), circ.get_all_out_edges(v), {v}};
221  circ.substitute(inner_circ, sub);
222  }
223  return !circbox_verts
224  .empty(); // always true if we have left Circuit formalism
225  });
226 }
