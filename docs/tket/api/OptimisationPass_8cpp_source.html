61 Transform synthesise_tket() {
62  Transform seq = commute_through_multis() >> remove_redundancies();
63  Transform rep = repeat(seq);
64  Transform synth = decompose_multi_qubits_CX() >> remove_redundancies() >>
65  rep >> squash_1qb_to_tk1();
66  Transform small_part = remove_redundancies() >> rep >> squash_1qb_to_tk1();
67  Transform repeat_synth = repeat_with_metric(
68  small_part, [](const Circuit &circ) { return circ.n_vertices(); });
69  return synth >> repeat_synth;
70 }

72 static Transform CXs_from_phase_gadgets(CXConfigType cx_config) {
73  return Transform([=](Circuit &circ) {
74  bool success = false;
75  VertexList bin;
76  auto [i, end] = boost::vertices(circ.dag);
77  for (auto next = i; i != end; i = next) {
78  ++next;
79  Vertex a = *i;
80  Op_ptr op = circ.get_Op_ptr_from_Vertex(a);
81  if (op->get_type() == OpType::PhaseGadget) {
82  unsigned n_qubits = op->n_qubits();
83  Circuit replacement =
84  phase_gadget(n_qubits, op->get_params()[0], cx_config);
85  Subcircuit sub = {
86  {circ.get_in_edges(a)}, {circ.get_all_out_edges(a)}, {a}};
87  circ.substitute(replacement, sub, Circuit::VertexDeletion::Yes);
88  success = true;
89  }
90  }
91  return success;
92  });
93 }

101 Transform synthesise_OQC() {

112 /* Returns a Circuit with only HQS allowed Ops (Rz, PhasedX, ZZMax) */
113 Transform synthesise_HQS() {
114  return Transform([](Circuit &circ) {
115  Transform single_loop =
116  remove_redundancies() >> commute_through_multis() >> reduce_XZ_chains();
117  Transform hqs_loop = remove_redundancies() >> commute_and_combine_HQS2() >>
118  reduce_XZ_chains();
119  Transform main_seq = decompose_multi_qubits_CX() >> clifford_simp() >>
120  decompose_ZX() >> repeat(single_loop) >>
121  decompose_CX_to_HQS2() >> repeat(hqs_loop) >>
122  decompose_ZX_to_HQS1();
123  return main_seq.apply(circ);
124  });
125 }
