31 bool Rewrite::spider_fusion_fun(ZXDiagram& diag) {
32  bool success = false;
33  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
34  ZXType vtype = diag.get_zxtype(v);
35  if (!is_spider_type(vtype)) continue;
42  WireVec adj_vec = diag.adj_wires(v);
43  std::list<Wire> adj_list{adj_vec.begin(), adj_vec.end()};
44  while (!adj_list.empty()) {
45  Wire w = adj_list.front();
46  adj_list.pop_front();
47  ZXWireType wtype = diag.get_wire_type(w);
48  ZXVert u = diag.other_end(w, v);
49  ZXType utype = diag.get_zxtype(u);
50  bool same_colour = vtype == utype;
51  if (!is_spider_type(utype) || u == v ||
52  (wtype == ZXWireType::Basic) != same_colour)
53  continue;
54  // The spiders `u` and `v` can be fused together
55  // We merge into `v` and remove `u` so that we can efficiently continue to
56  // search the neighbours
57  const BasicGen& vspid = diag.get_vertex_ZXGen<BasicGen>(v);
58  const BasicGen& uspid = diag.get_vertex_ZXGen<BasicGen>(u);
59  ZXGen_ptr new_spid = std::make_shared<const BasicGen>(
60  vtype, vspid.get_param() + uspid.get_param(),
61  (vspid.get_qtype() == QuantumType::Classical ||
62  uspid.get_qtype() == QuantumType::Classical)
63  ? QuantumType::Classical
64  : QuantumType::Quantum);
65  diag.set_vertex_ZXGen_ptr(v, new_spid);
66  for (const Wire& uw : diag.adj_wires(u)) {
67  WireEnd u_end = diag.end_of(uw, u);
68  ZXVert other = diag.other_end(uw, u);
69  WireProperties uwp = diag.get_wire_info(uw);
70  // Wires may need flipping type to match colours
71  if (!same_colour)
72  uwp.type = (uwp.type == ZXWireType::Basic) ? ZXWireType::H
73  : ZXWireType::Basic;
78  if (other == v && uwp.type == ZXWireType::Basic) continue;
79  // Self loops on `u` needs to become self loops on `v`
80  if (other == u) other = v;
81  // Connect edge to `v` instead with the same properties.
82  Wire new_w;
83  if (u_end == WireEnd::Source)
84  new_w = diag.add_wire(v, other, uwp);
85  else
86  new_w = diag.add_wire(other, v, uwp);
87  // Iteratively fuse along new wire if possible
88  adj_list.push_back(new_w);
89  }
90  // Remove `u`
91  diag.remove_vertex(u);
92  success = true;
93  }
94  }
95  return success;
96 }

139 bool Rewrite::parallel_h_removal_fun(ZXDiagram& diag) {
140  bool success = false;
141  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
142  ZXType vtype = diag.get_zxtype(v);
143  if (!is_spider_type(vtype)) continue;
144  QuantumType vqtype = *diag.get_qtype(v);
145  std::map<ZXVert, Wire> h_wires;
146  for (const Wire& w : diag.adj_wires(v)) {
147  ZXWireType wtype = diag.get_wire_type(w);
148  ZXVert u = diag.other_end(w, v);
149  ZXType utype = diag.get_zxtype(u);
150  if (!is_spider_type(utype)) continue;
151  if ((wtype == ZXWireType::H) != (utype == vtype)) continue;
152  // This is (effectively) a Hadamard edge
153  QuantumType uqtype = *diag.get_qtype(u);
154  QuantumType wqtype = diag.get_qtype(w);
155  if (vqtype == QuantumType::Classical &&
156  uqtype == QuantumType::Classical && wqtype == QuantumType::Quantum) {
157  // Doubled wire forms a pair
158  diag.remove_wire(w);
159  success = true;
160  continue;
161  }
162  // Look for another wire to pair it with
163  auto added = h_wires.insert({u, w});
164  if (!added.second) {
165  // Already found the other of the pair, so remove both
166  Wire other_w = added.first->second;
167  h_wires.erase(added.first);
168  diag.remove_wire(w);
169  diag.remove_wire(other_w);
170  success = true;
171  }
172  }
173  }
174  return success;
175 }
