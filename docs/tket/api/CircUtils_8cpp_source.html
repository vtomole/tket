32 Eigen::Matrix2cd get_matrix(const Circuit &circ, const Vertex &vert) {
33  const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
34  if (op->get_type() != OpType::tk1) {
35  throw NotImplemented(
36  "Cannot obtain matrix from gate: " + op->get_name() +
37  ". Try rebasing to tket's internal representation.");
38  }
39  std::vector<Expr> ps = op->get_params();
40  ps.push_back(0);
41  return get_matrix_from_tk1_angles(ps);
42 }

44 Eigen::Matrix2cd get_matrix_from_circ(const Circuit &circ) {
45  if (circ.n_qubits() != 1)
46  throw CircuitInvalidity(
47  "Getting Matrix: expected 1 qubit circuit, found " +
48  std::to_string(circ.n_qubits()));
49  Complex factor = std::exp(i_ * PI * eval_expr(circ.get_phase()).value());
50  VertexVec qpath = circ.qubit_path_vertices(circ.all_qubits()[0]);
51  unsigned N = qpath.size();
52  if (N == 2) return factor * Eigen::Matrix2cd::Identity();
53  Eigen::Matrix2cd m = get_matrix(circ, qpath[N - 2]);
54  for (unsigned x = N - 3; x >= 1; --x) {
55  m = m * get_matrix(circ, qpath[x]);
56  }
57  return factor * m;
58 }

60 Eigen::Matrix4cd get_matrix_from_2qb_circ(const Circuit &circ) {
61  std::vector<QPathDetailed> all_paths = circ.all_qubit_paths();
62  std::map<Vertex, Eigen::Matrix4cd> v_to_op;
63  Eigen::Matrix4cd cnot, tonc, swap;
64  // clang-format off
65  cnot << 1, 0, 0, 0,
66  0, 1, 0, 0,
67  0, 0, 0, 1,
68  0, 0, 1, 0;
69  tonc << 1, 0, 0, 0,
70  0, 0, 0, 1,
71  0, 0, 1, 0,
72  0, 1, 0, 0;
73  swap << 1, 0, 0, 0,
74  0, 0, 1, 0,
75  0, 1, 0, 0,
76  0, 0, 0, 1;
77  // clang-format on
78 
79  for (unsigned uqb = 0; uqb < 2; uqb++) {
80  for (QPathDetailed::iterator it = all_paths[uqb].begin();
81  it != all_paths[uqb].end(); ++it) {
82  const Op_ptr o = circ.get_Op_ptr_from_Vertex(it->first);
83  switch (o->get_type()) {
84  case OpType::Input:
85  case OpType::Create:
86  case OpType::Output:
87  case OpType::Discard: {
88  v_to_op[it->first] = Eigen::Matrix4cd::Identity();
89  break;
90  }
91  case OpType::SWAP: {
92  if (uqb == 0) v_to_op[it->first] = swap;
93  break;
94  }
95  case OpType::CX: {
96  if (uqb == 0) {
97  if (it->second == 0) {
98  v_to_op[it->first] = cnot;
99  } else {
100  v_to_op[it->first] = tonc;
101  }
102  }
103  break;
104  }
105  default: {
106  if (o->get_desc().is_gate() && circ.n_in_edges(it->first) == 1 &&
107  circ.n_out_edges(it->first) == 1) {
108  const Op_ptr g = o;
109  std::vector<Expr> ps = as_gate_ptr(g)->get_tk1_angles();
110  Eigen::Matrix2cd mat = get_matrix_from_tk1_angles(ps);
111  if (uqb == 0) {
112  v_to_op[it->first] =
113  Eigen::kroneckerProduct(mat, Eigen::Matrix2cd::Identity());
114  } else {
115  v_to_op[it->first] =
116  Eigen::kroneckerProduct(Eigen::Matrix2cd::Identity(), mat);
117  }
118  } else
119  throw NotImplemented(
120  "Cannot obtain matrix from op " + o->get_name());
121  }
122  }
123  }
124  }
125  Eigen::Matrix4cd m = Eigen::Matrix4cd::Identity();
126  SliceVec slices = circ.get_slices();
127  for (const Slice &s : slices) {
128  for (const Vertex &v : s) {
129  m = v_to_op[v] * m;
130  }
131  }
132  return std::exp(i_ * PI * eval_expr(circ.get_phase()).value()) * m;
133 }

136 Circuit two_qubit_canonical(const Eigen::Matrix4cd &U, double cx_fidelity) {
137  if (!is_unitary(U)) {
138  throw std::invalid_argument(
139  "Non-unitary matrix passed to two_qubit_canonical");
140  }
141 
142  auto [K1, A, K2] = get_information_content(U);
143 
144  K1 /= pow(K1.determinant(), 0.25);
145  K2 /= pow(K2.determinant(), 0.25);
146 
147  // Decompose Exp term
148  auto gates = expgate_as_CX(A, cx_fidelity);
149 
150  // Decompose single qubits
151  auto [K1a, K1b] = kronecker_decomposition(K1);
152  auto [K2a, K2b] = kronecker_decomposition(K2);
153 
154  Circuit result(2);
155 
156  if (gates.empty()) {
157  std::vector<double> angles_q0 = tk1_angles_from_unitary(K1a * K2a);
158  result.add_op<unsigned>(
159  OpType::tk1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
160  std::vector<double> angles_q1 = tk1_angles_from_unitary(K1b * K2b);
161  result.add_op<unsigned>(
162  OpType::tk1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
163  }
164  for (auto it = gates.begin(); it != gates.end(); ++it) {
165  auto [ga, gb] = *it;
166  if (it == gates.begin()) {
167  // begin of circuit -> merge with K2
168  ga = ga * K2a;
169  gb = gb * K2b;
170  }
171  if (it + 1 == gates.end()) {
172  // end of circuit -> merge with K1
173  ga = K1a * ga;
174  gb = K1b * gb;
175  }
176  std::vector<double> angles_q0 = tk1_angles_from_unitary(ga);
177  result.add_op<unsigned>(
178  OpType::tk1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
179  std::vector<double> angles_q1 = tk1_angles_from_unitary(gb);
180  result.add_op<unsigned>(
181  OpType::tk1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
182  if (it + 1 != gates.end()) {
183  result.add_op<unsigned>(OpType::CX, {0, 1});
184  }
185  }
186 
187  // this fixes phase if decomposition is exact
188  Eigen::Matrix4cd reminder = get_matrix_from_2qb_circ(result).adjoint() * U;
189  const Complex phase = reminder(0, 0); // reminder = phase * I
190  result.add_phase(arg(phase) / PI);
191  return result;
192 }

194 // Factorize U as VD where V corresponds to a 2-CX circuit and
195 // D = diag(z, z*, z*, z). Return V and z.
196 static std::pair<Eigen::Matrix4cd, Complex> decompose_VD(
197  const Eigen::Matrix4cd &U) {
198  if (!is_unitary(U)) {
199  throw std::invalid_argument("Non-unitary matrix passed to decompose_VD");
200  }
201 
202  // The calculations below are derived from the proof of Proposition V.2 in
203  // https://arxiv.org/abs/quant-ph/0308033.
204 
205  Eigen::Matrix4cd u = U / pow(U.determinant(), 0.25);
206  Complex a = u(3, 0) * u(0, 3) - u(2, 0) * u(1, 3) - u(1, 0) * u(2, 3) +
207  u(0, 0) * u(3, 3);
208  Complex b = u(3, 1) * u(0, 2) - u(2, 1) * u(1, 2) - u(1, 1) * u(2, 2) +
209  u(0, 1) * u(3, 2);
210  // Now we want to find z such that |z|=1 and (az* - bz) is real.
211  // The numerical stability of this function is a concern when a is close to
212  // -b*. This problem can be demonstrated in artificially constructed examples
213  // (passing unitaries very close to, but not quite, the identity to the
214  // functions below). In these cases the product VD (or DV) may not approximate
215  // U to within Eigen's default tolerance. Is there a way to dodge this issue?
216  Complex w = a + std::conj(b);
217  double d = std::abs(w);
218  // If w = 0 then we can set z = 1.
219  Complex z = (d < EPS) ? 1. : w / d;
220  Complex z0 = sqrt(z);
221  Complex z1 = std::conj(z0);
222  Eigen::Matrix4cd V = U;
223  V.col(0) *= z1;
224  V.col(1) *= z0;
225  V.col(2) *= z0;
226  V.col(3) *= z1;
227  return {V, z0};
228 }

230 std::pair<Circuit, Complex> decompose_2cx_VD(const Eigen::Matrix4cd &U) {
231  auto [V, z0] = decompose_VD(U);
232  Circuit circ = two_qubit_canonical(V);
233  TKET_ASSERT(circ.count_gates(OpType::CX) <= 2);
234  return {circ, z0};
235 }

237 std::pair<Circuit, Complex> decompose_2cx_DV(const Eigen::Matrix4cd &U) {
238  auto [V, z0] = decompose_VD(U.adjoint());
239  V.adjointInPlace();
240  Circuit circ = two_qubit_canonical(V);
241  TKET_ASSERT(circ.count_gates(OpType::CX) <= 2);
242  return {circ, std::conj(z0)};
243 }

245 Circuit phase_gadget(unsigned n_qubits, const Expr &t, CXConfigType cx_config) {
246  // Handle n_qubits==0 as a special case, or the calculations below
247  // go badly wrong.
248  Circuit new_circ(n_qubits);
249  if (n_qubits == 0) {
250  new_circ.add_phase(-t / 2);
251  return new_circ;
252  }
253  switch (cx_config) {
254  case CXConfigType::Snake: {
255  for (unsigned i = n_qubits - 1; i != 0; --i) {
256  unsigned j = i - 1;
257  new_circ.add_op<unsigned>(OpType::CX, {i, j});
258  }
259  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
260  for (unsigned i = 0; i != n_qubits - 1; ++i) {
261  unsigned j = i + 1;
262  new_circ.add_op<unsigned>(OpType::CX, {j, i});
263  }
264  break;
265  }
266  case CXConfigType::Star: {
267  for (unsigned i = n_qubits - 1; i != 0; --i) {
268  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
269  }
270  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
271  for (unsigned i = 1; i != n_qubits; ++i) {
272  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
273  }
274  break;
275  }
276  case CXConfigType::Tree: {
277  unsigned complete_layers = floor(log2(n_qubits));
278  unsigned dense_end = pow(2, complete_layers);
279  for (unsigned i = 0; i < n_qubits - dense_end; i++)
280  new_circ.add_op<unsigned>(
281  OpType::CX, {dense_end + i, dense_end - 1 - i});
282  for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
283  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
284  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
285  }
286  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
287  for (unsigned step_size = dense_end / 2; step_size >= 1; step_size /= 2) {
288  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
289  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
290  }
291  for (unsigned i = 0; i < n_qubits - dense_end; i++)
292  new_circ.add_op<unsigned>(
293  OpType::CX, {dense_end + i, dense_end - 1 - i});
294  break;
295  }
296  case CXConfigType::MultiQGate: {
297  std::vector<std::vector<unsigned>> conjugations;
298  int sign_correction = 1;
299  for (int q = n_qubits - 1; q > 0; q -= 2) {
300  if (q - 1 > 0) {
301  unsigned i = q, j = q - 1;
302  // this is only equal to the CX decompositions above
303  // up to phase, but phase differences are cancelled out by
304  // its dagger XXPhase(-1/2) below.
305  new_circ.add_op<unsigned>(OpType::H, {i});
306  new_circ.add_op<unsigned>(OpType::H, {j});
307  new_circ.add_op<unsigned>(OpType::XXPhase3, 0.5, {i, j, 0});
308  sign_correction *= -1;
309  conjugations.push_back({i, j, 0});
310  } else {
311  unsigned i = q;
312  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
313  conjugations.push_back({i, 0});
314  }
315  }
316  new_circ.add_op<unsigned>(OpType::Rz, sign_correction * t, {0});
317  for (const auto &conj : conjugations) {
318  if (conj.size() == 2) {
319  new_circ.add_op<unsigned>(OpType::CX, conj);
320  } else {
321  TKET_ASSERT(conj.size() == 3);
322  new_circ.add_op<unsigned>(OpType::XXPhase3, -0.5, conj);
323  new_circ.add_op<unsigned>(OpType::H, {conj[0]});
324  new_circ.add_op<unsigned>(OpType::H, {conj[1]});
325  }
326  }
327  break;
328  }
329  }
330  return new_circ;
331 }

333 Circuit pauli_gadget(
334  const std::vector<Pauli> &paulis, const Expr &t, CXConfigType cx_config) {
335  unsigned n = paulis.size();
336  Circuit circ(n);
337  std::vector<unsigned> qubits;
338  for (unsigned i = 0; i < n; i++) {
339  switch (paulis[i]) {
340  case Pauli::I:
341  break;
342  case Pauli::X:
343  circ.add_op<unsigned>(OpType::H, {i});
344  qubits.push_back(i);
345  break;
346  case Pauli::Y:
347  circ.add_op<unsigned>(OpType::V, {i});
348  qubits.push_back(i);
349  break;
350  case Pauli::Z:
351  qubits.push_back(i);
352  break;
353  }
354  }
355  if (qubits.empty()) {
356  circ.add_phase(-t / 2);
357  } else {
358  Vertex v = circ.add_op<unsigned>(OpType::PhaseGadget, t, qubits);
359  Circuit cx_gadget = phase_gadget(circ.n_in_edges(v), t, cx_config);
360  Subcircuit sub = {circ.get_in_edges(v), circ.get_all_out_edges(v), {v}};
361  circ.substitute(cx_gadget, sub, Circuit::VertexDeletion::Yes);
362  for (unsigned i = 0; i < n; i++) {
363  switch (paulis[i]) {
364  case Pauli::I:
365  break;
366  case Pauli::X:
367  circ.add_op<unsigned>(OpType::H, {i});
368  break;
369  case Pauli::Y:
370  circ.add_op<unsigned>(OpType::Vdg, {i});
371  break;
372  case Pauli::Z:
373  break;
374  }
375  }
376  }
377  return circ;
378 }

380 Circuit with_CX(Gate_ptr op) {
381  OpType optype = op->get_type();
382  std::vector<Expr> params = op->get_params();
383  unsigned n = op->n_qubits();
384  if (n == 0) {
385  return Circuit();
386  } else if (n == 1) {
387  Circuit c(1);
388  c.add_op(op, std::vector<unsigned>{0});
389  return c;
390  }
391  switch (optype) {
392  case OpType::CX: {
393  Circuit c(2);
394  c.add_op(op, std::vector<unsigned>{0, 1});
395  return c;
396  }
397  case OpType::CCX:
398  return CircPool::CCX_normal_decomp();
399  case OpType::CY:
400  return CircPool::CY_using_CX();
401  case OpType::CZ:
402  return CircPool::CZ_using_CX();
403  case OpType::CH:
404  return CircPool::CH_using_CX();
405  case OpType::CV:
406  return CircPool::CV_using_CX();
407  case OpType::CVdg:
408  return CircPool::CVdg_using_CX();
409  case OpType::CSX:
410  return CircPool::CSX_using_CX();
411  case OpType::CSXdg:
412  return CircPool::CSXdg_using_CX();
413  case OpType::CRz:
414  return CircPool::CRz_using_CX(params[0]);
415  case OpType::CRx:
416  return CircPool::CRx_using_CX(params[0]);
417  case OpType::CRy:
418  return CircPool::CRy_using_CX(params[0]);
419  case OpType::CU1:
420  return CircPool::CU1_using_CX(params[0]);
421  case OpType::CU3:
422  return CircPool::CU3_using_CX(params[0], params[1], params[2]);
423  case OpType::PhaseGadget:
424  return phase_gadget(n, params[0], CXConfigType::Snake);
425  case OpType::SWAP:
426  return CircPool::SWAP_using_CX_0();
427  case OpType::CSWAP:
428  return CircPool::CSWAP_using_CX();
429  case OpType::BRIDGE:
430  return CircPool::BRIDGE_using_CX_0();
431  case OpType::ECR:
432  return CircPool::ECR_using_CX();
433  case OpType::ISWAP:
434  return CircPool::ISWAP_using_CX(params[0]);
435  case OpType::ZZMax:
436  return CircPool::ZZMax_using_CX();
437  case OpType::XXPhase:
438  return CircPool::XXPhase_using_CX(params[0]);
439  case OpType::YYPhase:
440  return CircPool::YYPhase_using_CX(params[0]);
441  case OpType::ZZPhase:
442  return CircPool::ZZPhase_using_CX(params[0]);
443  case OpType::XXPhase3:
444  return CircPool::XXPhase3_using_CX(params[0]);
445  case OpType::ESWAP:
446  return CircPool::ESWAP_using_CX(params[0]);
447  case OpType::FSim:
448  return CircPool::FSim_using_CX(params[0], params[1]);
449  case OpType::Sycamore:
450  return CircPool::FSim_using_CX(1. / 2., 1. / 6.);
451  case OpType::ISWAPMax:
452  return CircPool::ISWAP_using_CX(1.);
453  case OpType::PhasedISWAP:
454  return CircPool::PhasedISWAP_using_CX(params[0], params[1]);
455  case OpType::NPhasedX:
456  return CircPool::NPhasedX_using_CX(n, params[0], params[1]);
457  default:
458  throw CircuitInvalidity("Cannot decompose " + op->get_name());
459  }

468 static Circuit CnU1(unsigned n_controls, Expr lambda) {
469  // CnU1(x) decomposes recursively as:
470  // C{n-1}U1(x/2)[ctrls]; U1(x/2)[tgt]; CnX; U1(-x/2)[tgt]; CnX
471  // We don't actually use recursion; just iterate starting with the first U1:
472  Circuit c(n_controls + 1);
473  Expr x = lambda / (1u << n_controls);
474  c.add_op<unsigned>(OpType::U1, x, {0});
475  std::vector<unsigned> cnx_qbs = {0};
476  for (unsigned i = 0; i < n_controls; i++) {
477  cnx_qbs.push_back(i + 1);
478  c.add_op<unsigned>(OpType::U1, x, {i + 1});
479  c.add_op<unsigned>(CNXTYPE(i + 2), cnx_qbs);
480  c.add_op<unsigned>(OpType::U1, -x, {i + 1});
481  c.add_op<unsigned>(CNXTYPE(i + 2), cnx_qbs);
482  x *= 2;
483  }
484  return c;
485 }

487 Circuit with_controls(const Circuit &c, unsigned n_controls) {
488  if (c.n_bits() != 0 || !c.is_simple()) {
489  throw CircuitInvalidity("Only default qubit register allowed");
490  }
491  if (c.has_implicit_wireswaps()) {
492  throw CircuitInvalidity("Circuit has implicit wireswaps");
493  }
494 
495  // Dispose of the trivial case
496  if (n_controls == 0) {
497  return c;
498  }
499 
500  unsigned c_n_qubits = c.n_qubits();
501 
502  // 1. Rebase to {CX, CCX, CnX, CnRy} and single-qubit gates
503  Circuit c1(c);
504  VertexList bin;
505  BGL_FORALL_VERTICES(v, c1.dag, DAG) {
506  Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
507  OpType optype = op->get_type();
508  if (is_gate_type(optype)) {
509  if (is_projective_type(optype)) {
510  throw CircuitInvalidity("Projective operations present");
511  }
512  if (is_box_type(optype)) {
513  throw CircuitInvalidity("Undecomposed boxes present");
514  }
515  if (is_single_qubit_type(optype)) {
516  continue;
517  }
518  if (optype == OpType::CX || optype == OpType::CCX ||
519  optype == OpType::CnX || optype == OpType::CnRy) {
520  continue;
521  }
522  Circuit replacement = with_CX(as_gate_ptr(op));
523  c1.substitute(replacement, v, Circuit::VertexDeletion::No);
524  bin.push_back(v);
525  }
526  }
527  c1.remove_vertices(
528  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
529 
530  // Capture the phase. We may adjust this during replacements below.
531  Expr a = c1.get_phase();
532 
533  // 2. Replace all gates with controlled versions
534  Circuit c2(n_controls + c_n_qubits);
535  for (Circuit::CommandIterator cit = c1.begin(); cit != c1.end(); ++cit) {
536  Op_ptr op = cit->get_op_ptr();
537  unit_vector_t args = cit->get_args();
538  unsigned n_args = args.size();
539  unsigned n_new_args = n_controls + n_args;
540  qubit_vector_t new_args(n_new_args);
541  for (unsigned i = 0; i < n_controls; i++) {
542  new_args[i] = Qubit(i);
543  }
544  for (unsigned i = 0; i < n_args; i++) {
545  new_args[n_controls + i] = Qubit(n_controls + args[i].index()[0]);
546  }
547  OpType optype = op->get_type();
548  std::vector<Expr> params = op->get_params();
549  switch (optype) {
550  case OpType::noop:
551  break;
552  case OpType::X:
553  case OpType::CX:
554  case OpType::CCX:
555  case OpType::CnX:
556  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
557  break;
558  case OpType::Ry:
559  case OpType::CnRy:
560  c2.add_op<Qubit>(OpType::CnRy, params, new_args);
561  break;
562  default: {
563  std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
564  Expr theta = tk1_angles[1];
565  Expr phi = tk1_angles[0] - 0.5;
566  Expr lambda = tk1_angles[2] + 0.5;
567  Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
568  // Operation is U3(theta, phi, lambda) + phase t.
569  // First absorb t in the overall phase.
570  a += t;
571  // Construct a multi-controlled U3, by extending the standard CU3-to-CX
572  // decomposition.
573  Qubit target = new_args[n_controls];
574  Circuit cnu1 = CnU1(n_controls - 1, 0.5 * (lambda + phi));
575  c2.append(cnu1);
576  c2.add_op<Qubit>(OpType::U1, 0.5 * (lambda - phi), {target});
577  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
578  c2.add_op<Qubit>(
579  OpType::U3, {-0.5 * theta, 0, -0.5 * (lambda + phi)}, {target});
580  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
581  c2.add_op<Qubit>(OpType::U3, {0.5 * theta, phi, 0}, {target});
582  } break;
583  }
584  }
585 
586  // 3. Account for phase by appending a CnU1 to the control qubits.
587  if (!equiv_0(a)) {
588  Circuit cnu1 = CnU1(n_controls - 1, a);
589  c2.append(cnu1);
590  }
591 
592  return c2;
593 }
