44 // breaks for devices with n_qubits >= UINT_MAX/2
45 // For internal use.
46 PathHandler::PathHandler(const MatrixXb &connectivity) {
47  unsigned n = connectivity.rows();
48  size = n;
49 
50  unsigned approx_infinity = UINT_MAX >> 1;
51  if (n >= approx_infinity) throw std::out_of_range("Qubit number too large");
52  distance_matrix_ = MatrixXu::Constant(n, n, approx_infinity);
53  path_matrix_ = MatrixXu::Constant(n, n, n); // set all unreachable nodes to n
54  connectivity_matrix_ = connectivity;
55 
56  // Floyd-Warshall with path reconstruction, see:
57  // https://en.wikipedia.org/wiki/Floydâ€“Warshall_algorithm#Pseudocode_[11]
58  for (unsigned i = 0; i != n; ++i) {
59  distance_matrix_(i, i) = 0;
60  path_matrix_(i, i) = i;
61  for (unsigned j = 0; j != n; ++j) {
62  if (i == j) continue;
63  if (connectivity_matrix_(i, j)) {
64  distance_matrix_(i, j) = 1;
65  path_matrix_(i, j) = j;
66  }
67  }
68  }
69  for (unsigned k = 0; k != n; ++k) {
70  for (unsigned i = 0; i != n; ++i) {
71  for (unsigned j = 0; j != n; ++j) {
72  unsigned threshold = distance_matrix_(i, k) + distance_matrix_(k, j);
73  if (distance_matrix_(i, j) > threshold) {
74  distance_matrix_(i, j) = threshold;
75  path_matrix_(i, j) = path_matrix_(i, k);
76  }
77  }
78  }
79  }
80 }

82 PathHandler PathHandler::construct_acyclic_handler() const {
83  PathHandler acyclic_handler;
84  unsigned n = distance_matrix_.rows();
85  MatrixXb acyclic_connectivity(n, n);
86  std::vector<unsigned> num_neighbours(n, 0);
87 
88  for (unsigned i = 0; i != n; ++i) {
89  for (unsigned j = 0; j != n; ++j) {
90  if (connectivity_matrix_(i, j)) ++num_neighbours[i];
91  }
92  }
93 
94  for (unsigned i = 0; i != n; ++i) {
95  for (unsigned j = 0; j != n; ++j) {
96  acyclic_connectivity(i, j) = 0;
97  }
98  }
99 
100  // find the centre
101  unsigned threshold_max_dist = n, max_dist_row = 0, centre_node = 0;
102 
103  for (unsigned i = 0; i != n; ++i) {
104  for (unsigned j = 0; j != n; ++j) {
105  if (distance_matrix_(i, j) > max_dist_row)
106  max_dist_row = distance_matrix_(i, j);
107  }
108  if (max_dist_row < threshold_max_dist) {
109  centre_node = i;
110  threshold_max_dist = max_dist_row;
111  }
112  max_dist_row = 0;
113  }
114 
115  // build the acyclic graph outwards from the centre
116  std::list<unsigned> current_layer_vertices{centre_node};
117  std::list<unsigned> next_layer_vertices;
118  std::vector<std::pair<unsigned, unsigned>> parents_neighbours(
119  n); // pair(num_neighbours, parent vertex)
120  std::vector<bool> vertices_in_tree(
121  n, 0); // track which vertices are in the acyclic graph
122  vertices_in_tree[centre_node] = 1;
123  std::pair<unsigned, unsigned> empty_pair{};
124  while (!current_layer_vertices.empty()) {
125  for (unsigned vert : current_layer_vertices) {
126  for (unsigned j = 0; j != n; ++j) {
127  if ((distance_matrix_(vert, j) == 1) & (!vertices_in_tree[j])) {
128  // if first encounter of vertex, add to next layer
129  if (parents_neighbours[j] == empty_pair) {
130  next_layer_vertices.push_back(j);
131  parents_neighbours[j] = {num_neighbours[vert], vert};
132 
133  }
134  // choose parent with most neighbours
135  else if (parents_neighbours[j].first < num_neighbours[vert]) {
136  parents_neighbours[j] = {num_neighbours[vert], vert};
137  }
138  }
139  }
140  }
141  current_layer_vertices.clear();
142  // add in edges
143  for (unsigned vert : next_layer_vertices) {
144  unsigned parent_vertex = parents_neighbours[vert].second;
145  acyclic_connectivity(vert, parent_vertex) = 1;
146  acyclic_connectivity(parent_vertex, vert) = 1;
147 
148  current_layer_vertices.push_back(vert);
149  vertices_in_tree[vert] = 1;
150  parents_neighbours[vert] = {};
151  }
152  next_layer_vertices.clear();
153  }
154 
155  return PathHandler(acyclic_connectivity);
156 }

167 std::vector<Node> find_hampath(const Architecture &arch, long timeout) {
168  using ArchitectureConn = Architecture::UndirectedConnGraph;
169  ArchitectureConn undirected_target = arch.get_undirected_connectivity();
170  unsigned n_nodes = arch.n_uids();
171 
172  std::vector<std::pair<Node, Node>> line_nodes(n_nodes - 1);
173  for (unsigned n = 0; n != n_nodes - 1; ++n) {
174  line_nodes[n] = {Node(n), Node(n + 1)};
175  }
176  Architecture line_arch(line_nodes);
177  ArchitectureConn undirected_pattern = line_arch.get_undirected_connectivity();
178  std::vector<qubit_bimap_t> all_maps;
179  vf2_match_add_callback<ArchitectureConn, ArchitectureConn> callback(
180  all_maps, undirected_pattern, undirected_target, 1);
181  bool found_monomorphism = boost::vf2_subgraph_mono(
182  undirected_pattern, undirected_target, callback, timeout);
183 
184  /* Architecture has no hampath, sad. */
185  if (!found_monomorphism) return {};
186 
187  /* Left: line, Right: input architecture. */
188  const qubit_bimap_t &qmap = all_maps[0];
189  std::vector<Node> hampath;
190  for (l_const_iterator_t it = qmap.left.begin(); it != qmap.left.end(); ++it) {
191  hampath.push_back(it->second);
192  }
193  return hampath;
194 }

196 IterationOrder::IterationOrder(const Architecture &arch) {
197  std::set<Node> visited_nodes;
198 
199  Node no = *arch.get_all_uids_set().begin();
200 
201  iterationorder.push_back(no);
202  visited_nodes.insert(no);
203 
204  unsigned whilecount = 0;
205  while ((visited_nodes.size() < arch.n_uids()) &&
206  (whilecount < arch.n_uids())) {
207  for (auto edge : arch.get_connections_vec()) {
208  if (visited_nodes.count(edge.first) &&
209  (!visited_nodes.count(edge.second))) {
210  iterationorder.push_back(edge.second);
211  visited_nodes.insert(edge.second);
212  edgelist.push_back(edge);
213  } else if (
214  (!visited_nodes.count(edge.first)) &&
215  (visited_nodes.count(edge.second))) {
216  iterationorder.push_back(edge.first);
217  visited_nodes.insert(edge.first);
218  edgelist.push_back(edge);
219  }
220  }
221  ++whilecount;
222  }
223 
224  if (visited_nodes.size() != arch.n_uids()) {
225  throw std::logic_error("Unconnected architecture");
226  }
227 
228  std::reverse(iterationorder.begin(), iterationorder.end());
229 }
