67 class PQPSquasher : public AbstractSquasher {
68  public:
69  PQPSquasher(
70  OpType p, OpType q, bool smart_squash = true, bool reversed = false)
71  : p_(p),
72  q_(q),
73  smart_squash_(smart_squash),
74  reversed_(reversed),
75  rotation_chain() {
76  if (!(p == OpType::Rx || p == OpType::Ry || p == OpType::Rz) ||
77  !(q == OpType::Rx || q == OpType::Ry || q == OpType::Rz)) {
78  throw std::logic_error(
79  "Can only reduce chains of single qubit rotations");
80  }
81  if (p == q) {
82  throw std::logic_error(
83  "Requires two different bases to perform single qubit "
84  "rotations");
85  }
86  }

88  bool accepts(OpType type) const override { return type == p_ || type == q_; }

92  throw NotValid("PQPSquasher: cannot append OpType");

97  std::pair<Circuit, Gate_ptr> flush(
98  std::optional<Pauli> commutation_colour = std::nullopt) const override {
99  bool commute_through = false;
100  OpType p = p_, q = q_;
101 
102  if (smart_squash_ && commutation_colour.has_value()) {
103  Gate P(p_, {0}, 1);
104  Gate Q(q_, {0}, 1);
105  if (P.commutes_with_basis(commutation_colour, 0)) {
106  commute_through = true;
107  } else if (Q.commutes_with_basis(commutation_colour, 0)) {
108  commute_through = true;
109  p = q_;
110  q = p_;
111  }
112  }
113 
114  // Construct list of merged rotations
115  std::list<Rotation> rots;
116  auto iter = rotation_chain.cbegin();
117  while (iter != rotation_chain.cend()) {
118  // Merge next q rotations
119  rots.push_back(merge_rotations(q, rotation_chain, iter));
120  // Merge next p rotations
121  rots.push_back(merge_rotations(p, rotation_chain, iter));
122  }
123 
124  // Perform any cancellations
125  std::list<Rotation>::iterator r = rots.begin();
126  while (r != rots.end()) {
127  if (r->is_id()) {
128  r = rots.erase(r);
129  if (r != rots.begin() && r != rots.end()) {
130  std::prev(r)->apply(*r);
131  r = rots.erase(r);
132  r--;
133  }
134  } else
135  r++;
136  }
137 
138  // Extract any P rotations from the beginning and end of the list
139  Expr p1 = 0, p2 = 0;
140  std::list<Rotation>::iterator i1 = rots.begin();
141  if (i1 != rots.end()) {
142  std::optional<Expr> a = i1->angle(p);
143  if (a) {
144  p1 = a.value();
145  rots.pop_front();
146  }
147  }
148  std::list<Rotation>::reverse_iterator i2 = rots.rbegin();
149  if (i2 != rots.rend()) {
150  std::optional<Expr> a = i2->angle(p);
151  if (a) {
152  p2 = a.value();
153  rots.pop_back();
154  }
155  }
156 
157  // Finish up:
158  Rotation R = {};
159  for (auto rot : rots) {
160  R.apply(rot);
161  }
162  std::tuple<Expr, Expr, Expr> angles = R.to_pqp(p, q);
163 
164  Expr angle_p1 = std::get<0>(angles) + p1;
165  Expr angle_q = std::get<1>(angles);
166  Expr angle_p2 = std::get<2>(angles) + p2;
167  fixup_angles(angle_p1, angle_q, angle_p2, reversed_);
168 
169  Circuit replacement(1);
170  Gate_ptr left_over_gate = nullptr;
171  if (!equiv_0(angle_p1, 4)) {
172  if (equiv_0(angle_q, 4) && equiv_0(angle_p2, 4) && commute_through) {
173  left_over_gate =
174  std::make_shared<Gate>(p, std::vector<Expr>{angle_p1}, 1);
175  } else {
176  replacement.add_op<unsigned>(p, angle_p1, {0});
177  }
178  }
179  if (!equiv_0(angle_q, 4)) {
180  replacement.add_op<unsigned>(q, angle_q, {0});
181  }
182  if (!equiv_0(angle_p2, 4)) {
183  if (commute_through) {
184  left_over_gate =
185  std::make_shared<Gate>(p, std::vector<Expr>{angle_p2}, 1);
186  } else {
187  replacement.add_op<unsigned>(p, angle_p2, {0});
188  }
189  }
190  redundancy_removal(replacement);
191  return {replacement, left_over_gate};
192  }

197  static Rotation merge_rotations(
198  OpType r, const std::vector<Gate_ptr> &chain,
199  std::vector<Gate_ptr>::const_iterator &iter) {
200  Expr total_angle(0);
201  while (iter != chain.end()) {
202  const Gate_ptr rot_op = *iter;
203  if (rot_op->get_type() != r) {
204  break;
205  }
206  total_angle += rot_op->get_params()[0];
207  iter++;
208  }
209  return Rotation(r, total_angle);
210  }

214  bool smart_squash_;

216  std::vector<Gate_ptr> rotation_chain;
217 };

243 static bool fixup_angles(
244  Expr &angle_p1, Expr &angle_q, Expr &angle_p2, bool reversed) {
245  bool success = false;
246  if (reversed) {
247  std::swap(angle_p1, angle_p2);
248  angle_p1 *= -1;
249  angle_q *= -1;
250  angle_p2 *= -1;
251  }
252  if (equiv_val(angle_q, 1., 2) && !equiv_0(angle_p2, 4)) {
253  // Prefer --P(p1-p2)--Q(...)--P(0)--
254  // Only occurs if angle_q is pi or 3pi and angle_p2 is non-zero
255  angle_p1 -= angle_p2;
256  angle_p2 = 0;
257  success = true;
258  } else if (equiv_val(angle_p2, 1., 4)) {
259  // Then prefer --P(p1+p2)--Q(-q)--P(0)--
260  // Only occurs if angle_p2 is pi
261  angle_p1 += 1;
262  angle_q *= -1;
263  angle_p2 = 0;
264  success = true;
265  } else if (equiv_val(angle_p2, 3., 4)) {
266  // Then prefer --P(p1+p2)--Q(-q)--P(0)--
267  // Only occurs if angle_p2 is 3pi
268  angle_p1 += 3;
269  angle_q *= -1;
270  angle_p2 = 0;
271  success = true;
272  } else if (equiv_val(angle_p1, 1., 4) && !equiv_0(angle_p2, 4)) {
273  // Then prefer --P(0)--Q(-q)--P(p1+p2)--
274  // Only occurs if angle_p1 is pi and angle_p2 is non-zero
275  angle_q *= -1;
276  angle_p2 += 1;
277  angle_p1 = 0;
278  success = true;
279  } else if (equiv_val(angle_p1, 3., 4) && !equiv_0(angle_p2, 4)) {
280  // Then prefer --P(0)--Q(-q)--P(p1+p2)--
281  // Only occurs if angle_p1 is 3pi and angle_p2 is non-zero
282  angle_q *= -1;
283  angle_p2 += 3;
284  angle_p1 = 0;
285  success = true;
286  }
287  if (reversed) {
288  std::swap(angle_p1, angle_p2);
289  angle_p1 *= -1;
290  angle_q *= -1;
291  angle_p2 *= -1;
292  }
293  return success;
294 }

296 static bool remove_redundancy(
297  Circuit &circ, const Vertex &vert, VertexList &bin,
298  std::set<IVertex> &new_affected_verts, IndexMap &im) {
299  const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
300  const OpDesc desc = op->get_desc();
301  if (!desc.is_gate()) return false;
302  if (circ.n_out_edges(vert) == 0 || circ.n_in_edges(vert) == 0) {
303  return false; // either a boundary vert or we have already removed it
304  }
305 
306  auto remove_single_vertex = [&bin, &circ, &new_affected_verts,
307  &im](const Vertex &v_remove) {
308  bin.push_back(v_remove);
309  for (const Vertex &l : circ.get_predecessors(v_remove)) {
310  new_affected_verts.insert({im.at(l), l});
311  }
312  circ.remove_vertex(
313  v_remove, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
314  };
315  // remove 0 angle rotations from circuit
316  std::optional<double> a = op->is_identity();
317  if (a) {
318  remove_single_vertex(vert);
319  circ.add_phase(a.value());
320  return true;
321  } else if (desc.type() == OpType::noop) {
322  // remove "noop" gates from circuit
323  remove_single_vertex(vert);
324  return true;
325  }
326  VertexVec kids = circ.get_successors(vert);
327 
328  // if op is immediately followed by a z basis measurement on all qubits,
329  // remove
330  if (circ.n_out_edges_of_type(vert, EdgeType::Classical) == 0) {
331  bool z_followed_by_measures = true;
332  for (port_t port = 0; port < kids.size() && z_followed_by_measures;
333  port++) {
334  if (circ.get_OpType_from_Vertex(kids[port]) == OpType::Measure) {
335  z_followed_by_measures &= op->commutes_with_basis(Pauli::Z, port);
336  } else {
337  z_followed_by_measures = false;
338  }
339  }
340  if (z_followed_by_measures) {
341  remove_single_vertex(vert);
342  return true;
343  }
344  }
345 
346  // check that both the vertex and its successor have each other and only each
347  // other
348  if ((kids.size() == 1) && (circ.get_predecessors(kids[0]).size() == 1)) {
349  // check that the ports match up between vertices
350  Vertex b = kids[0];
351  EdgeVec ins = circ.get_in_edges(b);
352  for (const Edge &in : ins) {
353  if (circ.get_source_port(in) != circ.get_target_port(in)) return false;
354  }
355 
356  // check that the classical edges match up correctly
357  if (circ.n_in_edges_of_type(vert, EdgeType::Boolean) != 0) return false;
358 
359  const Op_ptr b_op = circ.get_Op_ptr_from_Vertex(b);
360  const OpDesc b_desc = b_op->get_desc();
361 
362  if (!b_desc.is_oneway()) {
363  // if A = B.dagger(), AB = I
364  // This method cannot detect matches between rotation gates.
365  // Rotation gates are covered by the rotation gate combiner, everything
366  // else in this.
367  if (*b_op->dagger() == *op) {
368  bin.push_back(vert);
369  bin.push_back(b);
370  VertexVec last_verts = circ.get_predecessors(vert);
371  for (const Vertex &l : last_verts) {
372  new_affected_verts.insert({im.at(l), l});
373  }
374  VertexList to_detach{vert, b};
375  // detached from circuit but not removed from graph
376  circ.remove_vertices(
377  to_detach, Circuit::GraphRewiring::Yes,
378  Circuit::VertexDeletion::No);
379  return true;
380  } else if (desc.is_rotation()) {
381  // combine two rotation gates together, then if the combined
382  // operation is the identity up to phase, remove from circuit
383  if (b_desc.type() == desc.type()) {
384  Expr expr1 = op->get_params()[0];
385  Expr expr2 = b_op->get_params()[0];
386  VertexVec last_verts = circ.get_predecessors(vert);
387  for (const Vertex &l : last_verts) {
388  new_affected_verts.insert({im.at(l), l});
389  }
390  circ.remove_vertex(
391  b, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
392  bin.push_back(b);
393  std::vector<Expr> params_new = {expr1 + expr2};
394  Op_ptr op_new = get_op_ptr(desc.type(), params_new, ins.size());
395  std::optional<double> a = op_new->is_identity();
396  if (a) {
397  bin.push_back(vert);
398  circ.remove_vertex(
399  vert, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
400  circ.add_phase(a.value());
401  } else {
402  new_affected_verts.insert({im[vert], vert});
403  circ.dag[vert].op = op_new;
404  }
405  return true;
406  }
407  }
408  }
409  }
410  return false;
411 }

413 static bool redundancy_removal(Circuit &circ) {
414  bool success = false;
415  bool found_redundancy = true;
416  IndexMap im = circ.index_map();
417  std::set<IVertex> old_affected_verts;
418  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
419  old_affected_verts.insert({im.at(v), v});
420  }
421  VertexList bin;
422  while (found_redundancy) {
423  std::set<IVertex> new_affected_verts;
424  for (const IVertex &p : old_affected_verts) {
425  remove_redundancy(circ, p.second, bin, new_affected_verts, im);
426  }
427  found_redundancy = new_affected_verts.size() != 0;
428  success |= found_redundancy;
429  old_affected_verts = new_affected_verts;
430  }
431  circ.remove_vertices(
432  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
433  return success;
434 }
