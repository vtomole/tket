144 std::map<Bit, unsigned> Circuit::bit_readout() const {
145  std::map<Bit, unsigned> res;
146 
147  // Order bits to generate indices
148  bit_vector_t all_bs = all_bits();
149  std::sort(all_bs.begin(), all_bs.end());
150  unsigned i = 0;
151  for (const Bit &b : all_bs) {
152  res.insert({b, i});
153  i++;
154  }
155 
156  return res;
157 }

159 std::map<Qubit, unsigned> Circuit::qubit_readout() const {
160  std::map<Qubit, unsigned> res;
161  std::map<Bit, unsigned> bmap = bit_readout();
162 
163  // Find measurement map from qubits to index
164  for (auto [it, end] = boundary.get<TagType>().equal_range(UnitType::Qubit);
165  it != end; it++) {
166  Vertex q_out = it->out_;
167  Vertex last_gate = source(get_nth_in_edge(q_out, 0));
168  if (get_OpType_from_Vertex(last_gate) == OpType::Measure) {
169  Vertex possible_c_out = target(get_nth_out_edge(last_gate, 1));
170  if (get_OpType_from_Vertex(possible_c_out) == OpType::ClOutput) {
171  Bit b(get_id_from_out(possible_c_out));
172  res.insert({Qubit(it->id_), bmap.at(b)});
173  }
174  }
175  }
176 
177  return res;
178 }

279 // given a vertex, returns a set of all its successor vertices
280 // this set can be empty, and no warnings are given if it is
281 // there are no checks to ensure the vertex exists in the graph
282 VertexVec Circuit::get_successors(const Vertex &vert) const {
283  EdgeVec outs = get_all_out_edges(vert);
284  VertexVec children;
285  std::unordered_set<Vertex> lookup;
286  for (const Edge &e : outs) {
287  Vertex succ = target(e);
288  if (lookup.find(succ) == lookup.end()) {
289  children.push_back(succ);
290  lookup.insert(succ);
291  }
292  }
293  return children;
294 }

373 EdgeVec Circuit::get_in_edges(const Vertex &vert) const {
374  unsigned n = n_in_edges(vert);
375  EdgeVec inedges(n);
376  std::vector<bool> port_found(n, false);
377  BGL_FORALL_INEDGES(vert, e, dag, DAG) {
378  port_t port = get_target_port(e);
379  if (port >= n) {
380  inedges.resize(port + 1);
381  port_found.resize(port + 1, false);
382  } else if (port_found[port]) {
383  throw CircuitInvalidity("Vertex has multiple inputs on the same port");
384  }
385  port_found[port] = true;
386  inedges[port] = e;
387  }
388  for (unsigned i = 0; i < n; i++)
389  if (!port_found[i]) {
390  throw CircuitInvalidity("Input ports on Vertex are non-contiguous");
391  }
392  return inedges;
393 }

406 std::vector<std::optional<Edge>> Circuit::get_linear_out_edges(
407  const Vertex &vert) const {
408  unsigned n = n_ports(vert);
409  std::vector<std::optional<Edge>> outedges(n);
410  BGL_FORALL_OUTEDGES(vert, e, dag, DAG) {
411  if (get_edgetype(e) == EdgeType::Boolean) continue;
412  port_t port = get_source_port(e);
413  if (port >= n) {
414  throw CircuitInvalidity("Vertex has an output on an unexpected port");
415  }
416  if (outedges[port]) {
417  throw CircuitInvalidity(
418  "Vertex has multiple linear outputs on the same port");
419  }
420  outedges[port] = e;
421  }
422  return outedges;
423 }

425 EdgeVec Circuit::get_all_out_edges(const Vertex &vert) const {
426  std::vector<std::optional<Edge>> lin_outs = get_linear_out_edges(vert);
427  std::vector<EdgeVec> b_bundles = get_b_out_bundles(vert);
428  EdgeVec outs;
429  for (unsigned i = 0; i < lin_outs.size(); ++i) {
430  std::optional<Edge> &l_out = lin_outs[i];
431  if (l_out) {
432  outs.push_back(*l_out);
433  outs.insert(outs.end(), b_bundles[i].begin(), b_bundles[i].end());
434  }
435  }
436  return outs;
437 }

439 EdgeVec Circuit::get_out_edges_of_type(
440  const Vertex &vert, EdgeType type) const {
441  if (type == EdgeType::Boolean) {
442  std::vector<EdgeVec> bundles = get_b_out_bundles(vert);
443  EdgeVec outs;
444  for (const EdgeVec &b : bundles) {
445  outs.insert(outs.end(), b.begin(), b.end());
446  }
447  return outs;
448  } else {
449  std::vector<std::optional<Edge>> outs = get_linear_out_edges(vert);
450  EdgeVec matching;
451  for (const std::optional<Edge> &e : outs) {
452  if (e && get_edgetype(*e) == type) {
453  matching.push_back(*e);
454  }
455  }
456  return matching;
457  }

460 std::vector<EdgeVec> Circuit::get_b_out_bundles(const Vertex &vert) const {
461  unsigned n = n_ports(vert);
462  std::vector<EdgeVec> bundles(n);
463  BGL_FORALL_OUTEDGES(vert, e, dag, DAG) {
464  if (get_edgetype(e) == EdgeType::Boolean) {
465  port_t port = get_source_port(e);
466  if (port > n) {
467  throw CircuitInvalidity("Vertex has an output on an unexpected port");
468  }
469  bundles.at(port).push_back(e);
470  }
471  }
472  return bundles;
473 }
